# ADR-2117: Integration Test Strategy

**Status:** Proposed  
**Date:** 2026-07-21  
**Deciders:** Architecture Team  

---

## Context

EPIC 11 necesita tests de integración reales que验证 el flujo completo entre AI Core y Business Domain.

## Decision

Crear Integration Tests que usen base de datos real o mocks de repositorio.

## Test Categories

### 1. Gateway Integration Tests

```python
# tests/integration/ai_core/test_device_gateway_integration.py
class TestDeviceGatewayIntegration:
    """Tests para DeviceGateway con repositorio real."""
    
    @pytest.fixture
    def uow_factory(self):
        """UOW Factory con base de datos de test."""
        return create_test_uow_factory()
    
    @pytest.mark.asyncio
    async def test_get_device_returns_real_data(self, uow_factory):
        """Verifica que get_device retorna datos del repositorio."""
        gateway = DeviceGatewayImpl(uow_factory)
        device = await gateway.get_by_id("device-001", "tenant-001")
        
        assert device is not None
        assert device.id == "device-001"
        assert device.name == "Ventilator ICU-1"
    
    @pytest.mark.asyncio
    async def test_search_returns_filtered_results(self, uow_factory):
        """Verifica que search filtra correctamente."""
        gateway = DeviceGatewayImpl(uow_factory)
        devices = await gateway.search("ventilator", "tenant-001")
        
        assert len(devices) > 0
        assert all("ventilator" in d.name.lower() for d in devices)
```

### 2. Context Provider Integration Tests

```python
# tests/integration/ai_core/test_context_providers_integration.py
class TestDeviceContextProviderIntegration:
    """Tests para DeviceContextProvider con gateway real."""
    
    @pytest.fixture
    def provider(self):
        gateway = DeviceGatewayImpl(test_uow_factory())
        return DeviceContextProvider(device_gateway=gateway)
    
    @pytest.mark.asyncio
    async def test_get_context_returns_device_data(self, provider):
        """Verifica que provider usa gateway real."""
        query = ContextQuery(query="Status of ventilator ICU-1")
        items = await provider.get_context(query)
        
        assert len(items) > 0
        assert items[0].type == "device"
        assert items[0].data["name"] is not None
```

### 3. End-to-End Integration Tests

```python
# tests/integration/ai_core/test_e2e_integration.py
class TestAICoreE2EIntegration:
    """Tests end-to-end del flujo completo."""
    
    @pytest.mark.asyncio
    async def test_full_flow_user_to_response(self):
        """Verifica el flujo completo: usuario → respuesta."""
        controller = AICoreController()
        await controller.initialize()
        
        response = await controller.process(
            user_input="Show me devices needing maintenance",
            user_id="user-001",
            tenant_id="tenant-001"
        )
        
        # Verify response contains real data
        assert response.content is not None
        assert "ventilator" in response.content.lower() or \
               "maintenance" in response.content.lower()
```
