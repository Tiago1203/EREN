# ADR-2103: AI Tool Registration Strategy

**Status:** Accepted  
**Date:** 2026-07-20  
**Deciders:** Architecture Board  

---

## Context

El Tool Orchestrator necesita registrar y ejecutar herramientas de dominio. Cada tool debe:
- Mapearse a un Domain Gateway
- Registrarse automáticamente
- Ejecutarse de forma aislada

---

## Decision

Implementar estrategia de registro automático con auto-discovery.

### Estructura

```
core/ai/tools/domain/
├── __init__.py              # Auto-registration
├── base.py                  # BaseDomainTool
├── device_tools.py           # Device tools
├── incident_tools.py         # Incident tools
├── knowledge_tools.py        # Knowledge tools
├── recommendation_tools.py   # Recommendation tools
└── hospital_tools.py         # Hospital tools
```

### Contrato Base

```python
# base.py
from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import Any, TYPE_CHECKING

if TYPE_CHECKING:
    from core.PHASE_2.ai.tools import ToolDefinition, ToolResult

@dataclass
class DomainToolConfig:
    """Configuration for domain tools."""
    name: str
    description: str
    category: str
    parameters_schema: dict
    requires_tenant: bool = True

class BaseDomainTool(ABC):
    """
    Base class for all domain tools.
    
    Provides common functionality for tool execution,
    error handling, and result formatting.
    """
    
    config: DomainToolConfig
    
    @abstractmethod
    async def execute(
        self,
        parameters: dict,
        tenant_id: str,
    ) -> "ToolResult":
        """Execute the tool with given parameters."""
        raise NotImplementedError
    
    def _validate_parameters(
        self,
        parameters: dict,
        required: list[str],
    ) -> None:
        """Validate required parameters."""
        missing = [p for p in required if p not in parameters]
        if missing:
            raise ValueError(f"Missing required parameters: {missing}")
    
    async def _safe_execute(
        self,
        parameters: dict,
        tenant_id: str,
    ) -> "ToolResult":
        """Execute with error handling."""
        from core.PHASE_2.ai.tools import ToolResult, ToolStatus
        
        try:
            result = await self.execute(parameters, tenant_id)
            return result
        except Exception as e:
            return ToolResult(
                status=ToolStatus.ERROR,
                data=None,
                error=str(e),
            )
```

### Ejemplo: SearchDeviceTool

```python
# device_tools.py
from dataclasses import dataclass
from typing import TYPE_CHECKING

from .base import BaseDomainTool, DomainToolConfig
from core.PHASE_2.ai.domain import DeviceGateway, DeviceDTO

if TYPE_CHECKING:
    from core.PHASE_2.ai.tools import ToolDefinition, ToolResult, ToolStatus

@dataclass
class SearchDeviceToolConfig:
    """Configuration for SearchDeviceTool."""
    name = "search_device"
    description = "Search for devices by name, type, or serial number"
    category = "device"
    parameters_schema = {
        "type": "object",
        "properties": {
            "query": {
                "type": "string",
                "description": "Search query",
            },
            "status": {
                "type": "string",
                "enum": ["active", "inactive", "maintenance"],
            },
            "limit": {
                "type": "integer",
                "default": 10,
            },
        },
        "required": ["query"],
    }

class SearchDeviceTool(BaseDomainTool):
    """
    Tool for searching devices.
    
    Usage:
        search_device(query="ventilator", status="active")
    """
    
    config = SearchDeviceToolConfig()
    
    def __init__(self, device_gateway: DeviceGateway):
        self._gateway = device_gateway
    
    async def execute(
        self,
        parameters: dict,
        tenant_id: str,
    ) -> "ToolResult":
        """Search devices."""
        from core.PHASE_2.ai.tools import ToolResult, ToolStatus
        
        self._validate_parameters(parameters, ["query"])
        
        results = await self._gateway.search(
            query=parameters["query"],
            tenant_id=tenant_id,
            filters={"status": parameters.get("status")},
            limit=parameters.get("limit", 10),
        )
        
        return ToolResult(
            status=ToolStatus.SUCCESS,
            data=[self._device_to_dict(d) for d in results],
        )
    
    def _device_to_dict(self, device: DeviceDTO) -> dict:
        """Convert DeviceDTO to dict for tool result."""
        return {
            "id": device.id,
            "name": device.name,
            "device_type": device.device_type,
            "status": device.status,
            "is_critical": device.is_critical,
            "location": device.location,
        }
```

### Auto-Registration

```python
# __init__.py
"""
Domain Tools Module.

This module auto-registers all domain tools with the ToolOrchestrator.
"""

from typing import TYPE_CHECKING

if TYPE_CHECKING:
    from core.PHASE_2.ai.tools import ToolOrchestrator

def register_domain_tools(orchestrator: "ToolOrchestrator") -> None:
    """
    Register all domain tools with the orchestrator.
    
    Called during AI Core initialization.
    """
    from core.PHASE_2.ai.domain import (
        DeviceGateway,
        IncidentGateway,
        KnowledgeGateway,
        RecommendationGateway,
        HospitalGateway,
    )
    from .device_tools import SearchDeviceTool, GetDeviceHistoryTool
    from .incident_tools import SearchIncidentTool, AnalyzeIncidentTool
    from .knowledge_tools import SearchKnowledgeTool, SearchManualTool
    from .recommendation_tools import GenerateRecommendationTool
    from .hospital_tools import GetCapacityInfoTool, GetDepartmentInfoTool
    
    # Create gateways (would use DI in production)
    device_gateway = DeviceGateway()
    incident_gateway = IncidentGateway()
    knowledge_gateway = KnowledgeGateway()
    recommendation_gateway = RecommendationGateway()
    hospital_gateway = HospitalGateway()
    
    # Register Device tools
    orchestrator.register(
        tool=SearchDeviceTool(device_gateway).config,
        implementation=SearchDeviceTool(device_gateway),
    )
    orchestrator.register(
        tool=GetDeviceHistoryTool(device_gateway, incident_gateway).config,
        implementation=GetDeviceHistoryTool(device_gateway, incident_gateway),
    )
    
    # Register Incident tools
    orchestrator.register(
        tool=SearchIncidentTool(incident_gateway).config,
        implementation=SearchIncidentTool(incident_gateway),
    )
    orchestrator.register(
        tool=AnalyzeIncidentTool(incident_gateway, device_gateway).config,
        implementation=AnalyzeIncidentTool(incident_gateway, device_gateway),
    )
    
    # Register Knowledge tools
    orchestrator.register(
        tool=SearchKnowledgeTool(knowledge_gateway).config,
        implementation=SearchKnowledgeTool(knowledge_gateway),
    )
    orchestrator.register(
        tool=SearchManualTool(knowledge_gateway).config,
        implementation=SearchManualTool(knowledge_gateway),
    )
    
    # Register Recommendation tools
    orchestrator.register(
        tool=GenerateRecommendationTool(recommendation_gateway).config,
        implementation=GenerateRecommendationTool(recommendation_gateway),
    )
    
    # Register Hospital tools
    orchestrator.register(
        tool=GetCapacityInfoTool(hospital_gateway).config,
        implementation=GetCapacityInfoTool(hospital_gateway),
    )
    orchestrator.register(
        tool=GetDepartmentInfoTool(hospital_gateway).config,
        implementation=GetDepartmentInfoTool(hospital_gateway),
    )
```

---

## Consequences

### Positive
- Auto-registration reduce boilerplate
- Easy to add new tools
- Centralized tool configuration
- Consistent tool interface

### Negative
- Magic auto-registration puede ser difícil de debuggear
- Posible circular imports
