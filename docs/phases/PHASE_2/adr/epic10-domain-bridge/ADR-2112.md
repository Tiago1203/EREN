# ADR-2112: Event Bridge Pattern

**Status:** Proposed  
**Date:** 2026-07-20  
**Deciders:** Architecture Team  

---

## Context

El AI Core necesita comunicarse con el Business Domain a través de eventos:
- AI detecta situaciones que requieren acción del dominio
- AI necesita reaccionar a eventos del dominio
- Mantener coherencia entre estado de AI y Dominio

---

## Decision

Crear un **EventBridge** que conecte el AI Event System con el Event Bus del dominio.

### Estructura

```python
# core/ai/integration/event_bridge.py
from __future__ import annotations

from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import Callable, Any, TYPE_CHECKING

if TYPE_CHECKING:
    from core.events import EventBus


class AIEventType(str, Enum):
    """Tipos de eventos generados por el AI Core."""
    INCIDENT_DETECTED = "incident_detected"
    RECOMMENDATION_GENERATED = "recommendation_generated"
    DEVICE_STATUS_CHANGED = "device_status_changed"
    MAINTENANCE_RECOMMENDED = "maintenance_recommended"
    ANOMALY_DETECTED = "anomaly_detected"
    CONTEXT_REFRESHED = "context_refreshed"


@dataclass(frozen=True)
class AIEvent:
    """
    Evento generado por el AI Core.
    
    Estos eventos son publicados al Event Bus del dominio
    para que otras partes del sistema reaccionen.
    """
    event_type: AIEventType
    tenant_id: str
    user_id: str | None = None
    timestamp: datetime = field(default_factory=datetime.utcnow)
    data: dict = field(default_factory=dict)
    metadata: dict = field(default_factory=dict)
    
    def to_domain_event(self) -> dict:
        """Convierte a formato de evento del dominio."""
        return {
            "source": "ai_core",
            "type": self.event_type.value,
            "tenant_id": self.tenant_id,
            "user_id": self.user_id,
            "timestamp": self.timestamp.isoformat(),
            "data": self.data,
            "metadata": self.metadata,
        }


class DomainEventType(str, Enum):
    """Tipos de eventos del dominio que interesan a AI Core."""
    DEVICE_CREATED = "device.created"
    DEVICE_UPDATED = "device.updated"
    DEVICE_STATUS_CHANGED = "device.status_changed"
    INCIDENT_CREATED = "incident.created"
    INCIDENT_UPDATED = "incident.updated"
    INCIDENT_RESOLVED = "incident.resolved"
    RECOMMENDATION_CREATED = "recommendation.created"
    KNOWLEDGE_UPDATED = "knowledge.updated"


@dataclass
class EventSubscription:
    """Suscripción a un tipo de evento."""
    event_type: DomainEventType
    handler: Callable[[dict], Any]
    filter_tenant: str | None = None


class EventBridge:
    """
    Bridge entre AI Core Events y Domain Event Bus.
    
    Permite:
    - Publicar eventos de AI al Event Bus del dominio
    - Suscribir a eventos del dominio
    - Mantener coherencia
    
    Usage:
        bridge = EventBridge(event_bus)
        
        # Publicar evento de AI
        await bridge.publish_ai_event(AIEvent(
            event_type=AIEventType.INCIDENT_DETECTED,
            tenant_id="tenant-001",
            data={"device_id": "dev-001", "severity": "high"},
        ))
        
        # Suscribir a eventos del dominio
        await bridge.subscribe_to_domain_events()
    """
    
    def __init__(self, event_bus: EventBus):
        self._event_bus = event_bus
        self._subscriptions: list[EventSubscription] = []
        self._handlers: dict[DomainEventType, list[Callable]] = {}
    
    async def publish_ai_event(self, event: AIEvent) -> None:
        """
        Publica un evento del AI Core al Event Bus.
        
        Args:
            event: Evento a publicar
        """
        domain_event = event.to_domain_event()
        
        # Publicar al Event Bus
        await self._event_bus.publish(
            topic=f"ai.{event.event_type.value}",
            event=domain_event,
            tenant_id=event.tenant_id,
        )
    
    async def subscribe_to_domain_events(
        self,
        event_types: list[DomainEventType] | None = None,
    ) -> None:
        """
        Suscribe a eventos del dominio.
        
        Args:
            event_types: Tipos de evento a escuchar (None = todos)
        """
        types_to_subscribe = event_types or list(DomainEventType)
        
        for event_type in types_to_subscribe:
            await self._event_bus.subscribe(
                topic=f"domain.{event_type.value}",
                handler=self._create_handler(event_type),
            )
    
    def on_domain_event(
        self,
        event_type: DomainEventType,
    ) -> Callable:
        """
        Decorador para suscribir handlers a eventos del dominio.
        
        Usage:
            bridge = EventBridge(event_bus)
            
            @bridge.on_domain_event(DomainEventType.INCIDENT_CREATED)
            async def handle_incident(event_data):
                # Actualizar contexto con nuevo incidente
                await context_builder.refresh("incident", event_data["incident_id"])
        """
        def decorator(handler: Callable) -> Callable:
            self._register_handler(event_type, handler)
            return handler
        return decorator
    
    def _register_handler(
        self,
        event_type: DomainEventType,
        handler: Callable,
    ) -> None:
        """Registra un handler para un tipo de evento."""
        if event_type not in self._handlers:
            self._handlers[event_type] = []
        self._handlers[event_type].append(handler)
    
    async def _create_handler(
        self,
        event_type: DomainEventType,
    ) -> Callable[[dict], Any]:
        """Crea un handler para un tipo de evento."""
        async def handler(event_data: dict) -> None:
            handlers = self._handlers.get(event_type, [])
            for h in handlers:
                try:
                    await h(event_data)
                except Exception as e:
                    # Log error pero continuar con otros handlers
                    import logging
                    logging.error(f"Handler error for {event_type}: {e}")
        return handler
    
    async def publish_incident_detected(
        self,
        tenant_id: str,
        user_id: str | None,
        device_id: str,
        device_name: str,
        severity: str,
        description: str,
    ) -> None:
        """Helper para publicar incidente detectado."""
        await self.publish_ai_event(AIEvent(
            event_type=AIEventType.INCIDENT_DETECTED,
            tenant_id=tenant_id,
            user_id=user_id,
            data={
                "device_id": device_id,
                "device_name": device_name,
                "severity": severity,
                "description": description,
            },
        ))
    
    async def publish_recommendation_generated(
        self,
        tenant_id: str,
        user_id: str | None,
        recommendation_id: str,
        recommendation_type: str,
        confidence: float,
        actions: list[str],
    ) -> None:
        """Helper para publicar recomendación generada."""
        await self.publish_ai_event(AIEvent(
            event_type=AIEventType.RECOMMENDATION_GENERATED,
            tenant_id=tenant_id,
            user_id=user_id,
            data={
                "recommendation_id": recommendation_id,
                "type": recommendation_type,
                "confidence": confidence,
                "actions": actions,
            },
        ))
    
    async def publish_anomaly_detected(
        self,
        tenant_id: str,
        user_id: str | None,
        entity_type: str,
        entity_id: str,
        anomaly_type: str,
        details: dict,
    ) -> None:
        """Helper para publicar anomalía detectada."""
        await self.publish_ai_event(AIEvent(
            event_type=AIEventType.ANOMALY_DETECTED,
            tenant_id=tenant_id,
            user_id=user_id,
            data={
                "entity_type": entity_type,
                "entity_id": entity_id,
                "anomaly_type": anomaly_type,
                "details": details,
            },
        ))
```

### Ejemplo de Uso con Context Refresh

```python
# core/ai/integration/event_handlers.py
from core.ai.integration.event_bridge import EventBridge, DomainEventType
from core.ai.context_builder.builder import ContextBuilder

async def setup_ai_event_handlers(
    event_bridge: EventBridge,
    context_builder: ContextBuilder,
) -> None:
    """
    Configura handlers para eventos del dominio.
    
    Cuando ocurre un evento del dominio, actualiza el contexto
    del AI para mantener coherencia.
    """
    
    @event_bridge.on_domain_event(DomainEventType.DEVICE_STATUS_CHANGED)
    async def on_device_status_changed(event_data: dict) -> None:
        """Actualiza contexto cuando un dispositivo cambia de estado."""
        device_id = event_data.get("device_id")
        if device_id:
            # Invalidar cache de dispositivo en context
            await context_builder.invalidate(f"device:{device_id}")
    
    @event_bridge.on_domain_event(DomainEventType.INCIDENT_CREATED)
    async def on_incident_created(event_data: dict) -> None:
        """Actualiza contexto cuando se crea un incidente."""
        incident_id = event_data.get("incident_id")
        if incident_id:
            # Refrescar lista de incidentes
            await context_builder.refresh("incident", incident_id)
    
    @event_bridge.on_domain_event(DomainEventType.RECOMMENDATION_CREATED)
    async def on_recommendation_created(event_data: dict) -> None:
        """Actualiza contexto cuando se genera una recomendación."""
        recommendation_id = event_data.get("recommendation_id")
        if recommendation_id:
            await context_builder.refresh("recommendation", recommendation_id)
```

## Consequences

### Positive
- Comunicación bidireccional con dominio
- AI puede reaccionar a cambios del dominio
- Eventos de AI disponibles para otros sistemas
- Cohesión con el Event Bus existente

### Negative
- Coupling con Event Bus
- Posible duplicación de eventos
- Necesita gestión de errores robusta
