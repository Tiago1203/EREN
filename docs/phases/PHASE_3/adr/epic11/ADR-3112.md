## Decisión

### Version Schema

```python
@dataclass
class KnowledgeVersion:
    """Version of knowledge base."""
    version: str                    # e.g., "v2.3.0"
    major: int                      # Breaking changes
    minor: int                       # New knowledge
    patch: int                      # Bug fixes
    created_at: datetime
    changes: list[dict]
    approved_by: str
    notes: str
    knowledge_snapshot: dict         # Full snapshot


@dataclass
class GovernancePolicy:
    """Policy for governance."""
    policy_id: str
    name: str
    description: str
    required_approvers: list[str]
    min_quality_score: float
    max_impact_change: str          # "major", "minor", "patch"
    audit_required: bool


@dataclass
class ApprovalRecord:
    """Record of approval."""
    approval_id: str
    version: str
    approved_by: str
    approved_at: datetime
    decision: str                   # "approved", "rejected", "needs_review"
    comments: str
```

### Knowledge Version Manager

```python
class KnowledgeVersionManager:
    """Manages knowledge versions."""
    
    def __init__(self):
        self.versions: list[KnowledgeVersion] = []
        self.current_version = "v1.0.0"
    
    async def create_version(
        self,
        learning_package: LearningPackage,
        approval: ApprovalRecord,
    ) -> KnowledgeVersion:
        """Create new version based on changes."""
        
        # Determine version bump type
        change_type = self._determine_change_type(learning_package)
        
        # Calculate new version
        current = self._parse_version(self.current_version)
        
        if change_type == "major":
            new_version = f"v{current['major'] + 1}.0.0"
        elif change_type == "minor":
            new_version = f"v{current['major']}.{current['minor'] + 1}.0"
        else:
            new_version = f"v{current['major']}.{current['minor']}.{current['patch'] + 1}"
        
        # Create version
        version = KnowledgeVersion(
            version=new_version,
            major=current['major'] if change_type == "patch" else current['major'] + (1 if change_type == "major" else 0),
            minor=current['minor'] if change_type == "patch" else (current['minor'] + 1 if change_type == "minor" else 0),
            patch=0 if change_type != "patch" else current['patch'] + 1,
            created_at=datetime.now(),
            changes=self._extract_changes(learning_package),
            approved_by=approval.approved_by,
            notes=approval.comments,
            knowledge_snapshot=self._create_snapshot(learning_package),
        )
        
        self.versions.append(version)
        self.current_version = new_version
        
        return version
    
    def _determine_change_type(
        self,
        package: LearningPackage,
    ) -> str:
        """Determine type of version change."""
        
        # Major: Breaking changes (contradicts existing knowledge)
        if self._has_breaking_changes(package):
            return "major"
        
        # Minor: New knowledge additions
        if len(package.new_knowledge) > 0:
            return "minor"
        
        # Patch: Bug fixes, corrections
        return "patch"
    
    def _has_breaking_changes(self, package: LearningPackage) -> bool:
        """Check if package has breaking changes."""
        
        # Check for conflicting patterns
        for pattern in package.new_patterns:
            if pattern.confidence < 0.5:
                return True
        
        return False
```

### Model Governance

```python
class ModelGovernance:
    """Governs model evolution."""
    
    def __init__(self):
        self.policies: list[GovernancePolicy] = self._default_policies()
        self.approvals: list[ApprovalRecord] = []
    
    def _default_policies(self) -> list[GovernancePolicy]:
        """Default governance policies."""
        return [
            GovernancePolicy(
                policy_id="minor_updates",
                name="Minor Knowledge Updates",
                description="New patterns and confidence adjustments",
                required_approvers=["biomedical_engineer"],
                min_quality_score=0.75,
                max_impact_change="minor",
                audit_required=False,
            ),
            GovernancePolicy(
                policy_id="major_updates",
                name="Major Knowledge Updates",
                description="New rules or contradicting knowledge",
                required_approvers=["biomedical_engineer", "clinical_supervisor"],
                min_quality_score=0.85,
                max_impact_change="major",
                audit_required=True,
            ),
            GovernancePolicy(
                policy_id="critical_updates",
                name="Critical Updates",
                description="Safety rules or regulatory changes",
                required_approvers=["biomedical_engineer", "clinical_supervisor", "administrator"],
                min_quality_score=0.95,
                max_impact_change="major",
                audit_required=True,
            ),
        ]
    
    async def validate_approval(
        self,
        version: KnowledgeVersion,
        quality_report: QualityReport,
    ) -> ApprovalRecord:
        """Validate approval meets governance requirements."""
        
        # Select policy based on version
        policy = self._select_policy(version)
        
        # Check quality threshold
        if quality_report.scores.total_score < policy.min_quality_score:
            return ApprovalRecord(
                approval_id=f"approval_{datetime.now().timestamp()}",
                version=version.version,
                approved_by="SYSTEM",
                approved_at=datetime.now(),
                decision="rejected",
                comments=f"Quality score {quality_report.scores.total_score:.2f} below threshold {policy.min_quality_score}",
            )
        
        # Check required approvers
        # (In real implementation, this would check actual approvals)
        
        return ApprovalRecord(
            approval_id=f"approval_{datetime.now().timestamp()}",
            version=version.version,
            approved_by="governance_system",
            approved_at=datetime.now(),
            decision="approved",
            comments="Passed governance validation",
        )
```
