**Título:** Arquitectura del Motor de Mejora Continua

## Contexto

El Continuous Improvement Engine es el guardián de la evolución de EREN. Garantiza que todo conocimiento aprendido pase por validación antes de ser incorporado.

### Modelo Principal

```python
@dataclass
class KnowledgeRevision:
    """Revision of knowledge."""
    revision_id: str
    learning_package_id: str
    quality_score: float
    conflicts_detected: list[str]
    status: RevisionStatus
    reviewed_by: str | None
    reviewed_at: datetime | None


@dataclass
class KnowledgeVersion:
    """Version of knowledge base."""
    version: str
    created_at: datetime
    changes: list[dict]
    approved_by: str
    notes: str


@dataclass
class ImprovementResult:
    """Result of improvement process."""
    approved: bool
    new_version: str | None
    changes: list[dict]
    conflicts: list[str]
    warnings: list[str]


class ContinuousImprovementEngine:
    """Main continuous improvement engine."""
    
    def __init__(self):
        self.review_manager = LearningReviewManager()
        self.quality_analyzer = KnowledgeQualityAnalyzer()
        self.conflict_detector = ConflictDetectionEngine()
        self.expert_reviewer = ExpertReviewWorkflow()
        self.version_manager = KnowledgeVersionManager()
        self.rule_promoter = RulePromotionEngine()
        self.governance = ModelGovernance()
        self.rollback_manager = RollbackManager()
        self.performance_monitor = PerformanceMonitor()
        self.pipeline = EvolutionPipeline()
    
    async def improve(
        self,
        learning_package: LearningPackage,
    ) -> ImprovementResult:
        """Process learning package through improvement pipeline."""
        
        # 1. Create revision
        revision = await self.review_manager.create_revision(
            learning_package
        )
        
        # 2. Analyze quality
        quality_report = await self.quality_analyzer.analyze(
            learning_package
        )
        
        # 3. Detect conflicts
        conflicts = await self.conflict_detector.detect(
            learning_package
        )
        
        # 4. Check if auto-approve possible
        if self._can_auto_approve(quality_report, conflicts):
            # Auto-approve high quality, no conflicts
            return await self._auto_approve(learning_package)
        
        # 5. Require expert review
        review_result = await self.expert_reviewer.review(
            learning_package, quality_report, conflicts
        )
        
        if not review_result.approved:
            return ImprovementResult(
                approved=False,
                new_version=None,
                changes=[],
                conflicts=conflicts,
                warnings=["Rejected by expert review"],
            )
        
        # 6. Create new version
        new_version = await self.version_manager.create_version(
            learning_package, review_result
        )
        
        # 7. Update all engines
        await self.pipeline.update_all_engines(new_version)
        
        return ImprovementResult(
            approved=True,
            new_version=new_version.version,
            changes=learning_package.new_knowledge,
            conflicts=conflicts,
            warnings=[],
        )
```
