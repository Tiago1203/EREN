## Decisión

### Rollback Types

```python
class RollbackTrigger(Enum):
    """Triggers for rollback."""
    MANUAL = "manual"                    # Admin initiated
    AUTOMATIC = "automatic"              # Metric degradation
    SCHEDULED = "scheduled"             # Deployment failure
    EMERGENCY = "emergency"             # Critical safety issue


@dataclass
class RollbackOperation:
    """Rollback operation record."""
    rollback_id: str
    trigger: RollbackTrigger
    from_version: str
    to_version: str
    initiated_by: str
    reason: str
    started_at: datetime
    completed_at: datetime | None
    status: str                         # "in_progress", "completed", "failed"
    changes_reverted: list[str]


@dataclass
class PerformanceMetric:
    """Performance metric snapshot."""
    metric_name: str
    value: float
    threshold: float
    timestamp: datetime


@dataclass
class RollbackPolicy:
    """Policy for rollback decisions."""
    policy_id: str
    metric_name: str
    threshold: float
    comparison: str                     # "gt", "lt", "eq"
    window_minutes: int                 # Time window to consider
```

### Rollback Manager

```python
class RollbackManager:
    """Manages rollback operations."""
    
    def __init__(self):
        self.operations: list[RollbackOperation] = []
        self.policies: list[RollbackPolicy] = self._default_policies()
        self.version_manager = KnowledgeVersionManager()
    
    def _default_policies(self) -> list[RollbackPolicy]:
        """Default rollback policies."""
        return [
            RollbackPolicy(
                policy_id="accuracy_drop",
                metric_name="recommendation_accuracy",
                threshold=0.05,  # 5% drop
                comparison="lt",
                window_minutes=60,
            ),
            RollbackPolicy(
                policy_id="success_rate_drop",
                metric_name="success_rate",
                threshold=0.85,  # Below 85%
                comparison="lt",
                window_minutes=30,
            ),
            RollbackPolicy(
                policy_id="false_positive_increase",
                metric_name="false_positive_rate",
                threshold=0.15,  # Above 15%
                comparison="gt",
                window_minutes=60,
            ),
        ]
    
    async def check_rollback_needed(
        self,
        current_metrics: list[PerformanceMetric],
    ) -> bool:
        """Check if rollback should be triggered."""
        
        for metric in current_metrics:
            # Find applicable policy
            policy = self._find_policy(metric.metric_name)
            
            if not policy:
                continue
            
            # Check if threshold breached
            if self._check_threshold(metric, policy):
                return True
        
        return False
    
    def _check_threshold(
        self,
        metric: PerformanceMetric,
        policy: RollbackPolicy,
    ) -> bool:
        """Check if metric breaches policy threshold."""
        
        if policy.comparison == "gt":
            return metric.value > policy.threshold
        elif policy.comparison == "lt":
            return metric.value < policy.threshold
        else:
            return metric.value == policy.threshold
    
    async def execute_rollback(
        self,
        trigger: RollbackTrigger,
        reason: str,
        initiated_by: str,
    ) -> RollbackOperation:
        """Execute rollback to previous version."""
        
        # Get current and target versions
        current = self.version_manager.current_version
        previous = self._get_previous_version(current)
        
        if not previous:
            raise ValueError("No previous version available for rollback")
        
        # Create operation record
        operation = RollbackOperation(
            rollback_id=f"rb_{datetime.now().timestamp()}",
            trigger=trigger,
            from_version=current,
            to_version=previous,
            initiated_by=initiated_by,
            reason=reason,
            started_at=datetime.now(),
            completed_at=None,
            status="in_progress",
            changes_reverted=[],
        )
        
        self.operations.append(operation)
        
        # Execute rollback (in real implementation, this would:
        # 1. Load previous version snapshot
        # 2. Update all engines
        # 3. Verify system stability
        # 4. Mark operation complete)
        
        operation.completed_at = datetime.now()
        operation.status = "completed"
        
        return operation
    
    def _get_previous_version(self, current: str) -> str | None:
        """Get previous version for rollback."""
        
        # Parse current version
        parts = current.replace("v", "").split(".")
        major, minor, patch = int(parts[0]), int(parts[1]), int(parts[2])
        
        # Determine rollback target
        if patch > 0:
            return f"v{major}.{minor}.{patch - 1}"
        elif minor > 0:
            return f"v{major}.{minor - 1}.0"
        elif major > 1:
            return f"v{major - 1}.0.0"
        
        return None
    
    async def get_rollback_history(self) -> list[RollbackOperation]:
        """Get history of rollback operations."""
        return sorted(
            self.operations,
            key=lambda x: x.started_at,
            reverse=True
        )
```
