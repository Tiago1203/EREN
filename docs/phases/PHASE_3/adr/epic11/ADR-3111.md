### Quality Dimensions

```python
class QualityDimension(Enum):
    """Dimensions of knowledge quality."""
    ACCURACY = "accuracy"          # Is it correct?
    CONSISTENCY = "consistency"     # Does it conflict?
    EVIDENCE = "evidence"           # Is it supported?
    REPEATABILITY = "repeatability" # Can it be reproduced?
    COVERAGE = "coverage"           # Is it complete?
    IMPACT = "impact"              # What changes if adopted?


@dataclass
class QualityScore:
    """Quality score breakdown."""
    accuracy: float
    consistency: float
    evidence: float
    repeatability: float
    coverage: float
    impact: float
    total_score: float
    passed: bool
    recommendations: list[str]


@dataclass
class QualityReport:
    """Complete quality analysis report."""
    report_id: str
    knowledge_id: str
    scores: QualityScore
    dimension_weights: dict
    analysis_timestamp: datetime
    automated: bool
```

### Knowledge Quality Analyzer

```python
class KnowledgeQualityAnalyzer:
    """Analyzes quality of new knowledge."""
    
    # Minimum scores for approval
    MIN_ACCURACY = 0.80
    MIN_CONSISTENCY = 0.75
    MIN_EVIDENCE = 0.70
    MIN_TOTAL = 0.75
    
    async def analyze(
        self,
        learning_package: LearningPackage,
    ) -> QualityReport:
        """Analyze quality of learning package."""
        
        scores = QualityScore(
            accuracy=self._analyze_accuracy(learning_package),
            consistency=self._analyze_consistency(learning_package),
            evidence=self._analyze_evidence(learning_package),
            repeatability=self._analyze_repeatability(learning_package),
            coverage=self._analyze_coverage(learning_package),
            impact=self._analyze_impact(learning_package),
            total_score=0.0,
            passed=False,
            recommendations=[],
        )
        
        # Calculate total score
        scores.total_score = self._calculate_total(scores)
        
        # Determine if passed
        scores.passed = self._determine_pass(scores)
        
        # Generate recommendations
        scores.recommendations = self._generate_recommendations(scores)
        
        return QualityReport(
            report_id=f"qr_{datetime.now().timestamp()}",
            knowledge_id=learning_package.learning_metrics.get("id", ""),
            scores=scores,
            dimension_weights=self._default_weights(),
            analysis_timestamp=datetime.now(),
            automated=True,
        )
    
    def _analyze_accuracy(
        self,
        package: LearningPackage,
    ) -> float:
        """Analyze accuracy of knowledge."""
        
        # Check confidence scores
        confidence_updates = package.updated_confidence
        
        if not confidence_updates:
            return 0.5  # No data
        
        # Calculate accuracy based on confidence trends
        avg_confidence = sum(
            u.new_confidence for u in confidence_updates
        ) / len(confidence_updates)
        
        return min(1.0, avg_confidence)
    
    def _analyze_consistency(
        self,
        package: LearningPackage,
    ) -> float:
        """Analyze consistency with existing knowledge."""
        
        # Check for conflicts
        patterns = package.new_patterns
        
        if not patterns:
            return 0.7  # No new patterns, assume consistent
        
        # Calculate consistency score
        consistent_count = sum(
            1 for p in patterns if p.confidence >= 0.7
        )
        
        return consistent_count / len(patterns) if patterns else 1.0
    
    def _analyze_evidence(
        self,
        package: LearningPackage,
    ) -> float:
        """Analyze evidence supporting knowledge."""
        
        experience = package.experience_records
        
        if not experience:
            return 0.5  # No experience
        
        # More experience = better evidence
        experience_factor = min(len(experience) / 10, 1.0)
        
        return 0.5 + (experience_factor * 0.5)
    
    def _analyze_repeatability(
        self,
        package: LearningPackage,
    ) -> float:
        """Analyze repeatability of findings."""
        
        # Check if patterns appear multiple times
        patterns = package.new_patterns
        
        if not patterns:
            return 0.6
        
        # More similar cases = more repeatable
        similar = package.similar_cases
        
        return min(1.0, len(similar) / 5 + 0.5)
    
    def _analyze_coverage(
        self,
        package: LearningPackage,
    ) -> float:
        """Analyze coverage of domain."""
        
        # Check how many areas are covered
        patterns = package.new_patterns
        knowledge = package.new_knowledge
        
        total_items = len(patterns) + len(knowledge)
        
        # More items = better coverage (up to a point)
        return min(1.0, total_items / 10)
    
    def _analyze_impact(
        self,
        package: LearningPackage,
    ) -> float:
        """Analyze potential impact of knowledge."""
        
        # Check knowledge updates
        knowledge = package.new_knowledge
        
        if not knowledge:
            return 0.5
        
        # More updates = higher potential impact
        return min(1.0, len(knowledge) / 5)
    
    def _calculate_total(self, scores: QualityScore) -> float:
        """Calculate weighted total score."""
        weights = self._default_weights()
        
        return (
            scores.accuracy * weights["accuracy"] +
            scores.consistency * weights["consistency"] +
            scores.evidence * weights["evidence"] +
            scores.repeatability * weights["repeatability"] +
            scores.coverage * weights["coverage"] +
            scores.impact * weights["impact"]
        )
    
    def _default_weights(self) -> dict:
        """Default dimension weights."""
        return {
            "accuracy": 0.30,
            "consistency": 0.20,
            "evidence": 0.20,
            "repeatability": 0.10,
            "coverage": 0.10,
            "impact": 0.10,
        }
    
    def _determine_pass(self, scores: QualityScore) -> bool:
        """Determine if knowledge passes quality check."""
        return (
            scores.total_score >= self.MIN_TOTAL and
            scores.accuracy >= self.MIN_ACCURACY and
            scores.consistency >= self.MIN_CONSISTENCY and
            scores.evidence >= self.MIN_EVIDENCE
        )
```
