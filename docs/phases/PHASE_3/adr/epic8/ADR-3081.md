## Contexto

Validación de que la evidencia supporting una recomendación es suficiente, relevante y actual.

## Decisión

### Validación de Evidencia

```python
@dataclass
class EvidenceValidationResult:
    """Result of evidence validation."""
    is_sufficient: bool
    is_relevant: bool
    is_current: bool
    quality_score: float
    sufficiency_score: float
    reason: str | None
    missing_evidence: list[str]


class EvidenceValidator:
    """Validates evidence for recommendations."""
    
    # Thresholds
    MIN_EVIDENCE_COUNT = 3
    MIN_QUALITY_SCORE = 0.4
    MIN_RELEVANCE_SCORE = 0.5
    MAX_EVIDENCE_AGE_DAYS = 365
    
    async def validate(
        self,
        recommendation: Recommendation,
        context: dict,
    ) -> EvidenceValidationResult:
        """Validate evidence sufficiency and quality."""
        
        evidence = context.get("evidence_bundle", {})
        supporting = evidence.get("supporting", [])
        contradicting = evidence.get("contradicting", [])
        
        # Check quantity
        sufficient_count = len(supporting) >= self.MIN_EVIDENCE_COUNT
        
        # Check quality
        quality_scores = [e.get("quality_score", 0) for e in supporting]
        avg_quality = sum(quality_scores) / len(quality_scores) if quality_scores else 0
        sufficient_quality = avg_quality >= self.MIN_QUALITY_SCORE
        
        # Check relevance
        relevance_scores = [e.get("relevance_score", 0) for e in supporting]
        avg_relevance = sum(relevance_scores) / len(relevance_scores) if relevance_scores else 0
        sufficient_relevance = avg_relevance >= self.MIN_RELEVANCE_SCORE
        
        # Check for contradictions
        has_contradictions = len(contradicting) > 0
        contradiction_ratio = len(contradicting) / max(len(supporting), 1)
        
        # Check evidence age
        is_current = True
        for e in supporting:
            age_days = (datetime.now() - e.get("date", datetime.now())).days
            if age_days > self.MAX_EVIDENCE_AGE_DAYS:
                is_current = False
                break
        
        # Calculate sufficiency
        sufficiency_score = self._calculate_sufficiency(
            count=len(supporting),
            quality=avg_quality,
            relevance=avg_relevance,
            contradictions=contradiction_ratio,
        )
        
        # Determine if sufficient
        is_sufficient = (
            sufficient_count and
            sufficient_quality and
            sufficient_relevance and
            contradiction_ratio < 0.3
        )
        
        # Generate missing evidence list
        missing = []
        if not sufficient_count:
            missing.append(f"Need {self.MIN_EVIDENCE_COUNT - len(supporting)} more evidence items")
        if not sufficient_quality:
            missing.append("Evidence quality too low")
        if not sufficient_relevance:
            missing.append("Evidence not relevant to hypothesis")
        
        return EvidenceValidationResult(
            is_sufficient=is_sufficient,
            is_relevant=sufficient_relevance,
            is_current=is_current,
            quality_score=avg_quality,
            sufficiency_score=sufficiency_score,
            reason=None if is_sufficient else f"Insufficient: {', '.join(missing)}",
            missing_evidence=missing,
        )
    
    def _calculate_sufficiency(
        self,
        count: int,
        quality: float,
        relevance: float,
        contradictions: float,
    ) -> float:
        """Calculate evidence sufficiency score (0-1)."""
        # Count factor (0-1)
        count_factor = min(count / self.MIN_EVIDENCE_COUNT, 1.0)
        
        # Quality factor
        quality_factor = quality
        
        # Relevance factor
        relevance_factor = relevance
        
        # Contradiction penalty
        contradiction_penalty = contradictions * 0.3
        
        # Weighted combination
        score = (
            count_factor * 0.25 +
            quality_factor * 0.35 +
            relevance_factor * 0.30 -
            contradiction_penalty
        )
        
        return max(0.0, min(1.0, score))
```
