# ADR-3082: Consistency Checking Design

**Título:** Diseño de Verificación de Consistencia
**Estado:** ✅ COMPLETO
**Fecha:** 2026-07-21
**Decisor:** Architecture Team

---

## Contexto

Verificación de que las recomendaciones son consistentes internamente y externamente con conocimiento previo.

---

## Decisión

### Verificación de Consistencia

```python
@dataclass
class ConsistencyResult:
    """Result of consistency checking."""
    is_consistent: bool
    internal_consistent: bool
    external_consistent: bool
    inconsistencies: list[str]
    suggestions: list[str]
    adjustment: float | None  # Confidence adjustment


class ConsistencyChecker:
    """Checks consistency of recommendations."""
    
    async def check(
        self,
        recommendation: Recommendation,
        context: dict,
    ) -> ConsistencyResult:
        """Check internal and external consistency."""
        
        inconsistencies = []
        suggestions = []
        
        # Internal consistency
        internal = self._check_internal_consistency(
            recommendation, context
        )
        if not internal.is_consistent:
            inconsistencies.extend(internal.issues)
            suggestions.extend(internal.suggestions)
        
        # External consistency
        external = await self._check_external_consistency(
            recommendation, context
        )
        if not external.is_consistent:
            inconsistencies.extend(external.issues)
            suggestions.extend(external.suggestions)
        
        # Calculate adjustment
        adjustment = self._calculate_adjustment(
            internal.is_consistent,
            external.is_consistent,
            len(inconsistencies),
        )
        
        return ConsistencyResult(
            is_consistent=internal.is_consistent and external.is_consistent,
            internal_consistent=internal.is_consistent,
            external_consistent=external.is_consistent,
            inconsistencies=inconsistencies,
            suggestions=suggestions,
            adjustment=adjustment,
        )
    
    def _check_internal_consistency(
        self,
        recommendation: Recommendation,
        context: dict,
    ) -> ConsistencyResult:
        """Check internal consistency."""
        issues = []
        suggestions = []
        
        # Check hypothesis-evidence consistency
        hypothesis = context.get("hypothesis", {})
        evidence_bundle = context.get("evidence_bundle", {})
        
        # If hypothesis says "sensor failure" but evidence is about "power supply"
        hypothesis_type = hypothesis.get("type")
        evidence_types = [e.get("type") for e in evidence_bundle.get("supporting", [])]
        
        if hypothesis_type and evidence_types:
            # Check for type mismatch
            mismatch_count = sum(1 for et in evidence_types if et != hypothesis_type)
            mismatch_ratio = mismatch_count / len(evidence_types)
            
            if mismatch_ratio > 0.5:
                issues.append("Evidence type mismatch with hypothesis")
                suggestions.append("Review hypothesis or gather matching evidence")
        
        # Check recommendation-context consistency
        recommendation_type = recommendation.get("action_type")
        context_type = context.get("equipment_type")
        
        if recommendation_type and context_type:
            # Some recommendations are invalid for certain equipment
            incompatible = self._check_compatibility(recommendation_type, context_type)
            if incompatible:
                issues.append(f"Incompatible recommendation for {context_type}")
                suggestions.append(f"Use {incompatible} instead")
        
        return ConsistencyResult(
            is_consistent=len(issues) == 0,
            internal_consistent=len(issues) == 0,
            external_consistent=True,
            inconsistencies=issues,
            suggestions=suggestions,
            adjustment=None,
        )
    
    async def _check_external_consistency(
        self,
        recommendation: Recommendation,
        context: dict,
    ) -> ConsistencyResult:
        """Check external consistency with historical data."""
        issues = []
        suggestions = []
        
        # Check against historical cases
        similar_cases = context.get("similar_cases", [])
        
        if similar_cases:
            # Compare recommendation with similar successful cases
            success_rate = sum(1 for c in similar_cases if c.get("outcome") == "success")
            success_ratio = success_rate / len(similar_cases)
            
            if success_ratio < 0.7:
                issues.append(
                    f"Only {success_ratio:.0%} of similar cases succeeded"
                )
                suggestions.append(
                    "Review similar cases or gather more evidence"
                )
        
        return ConsistencyResult(
            is_consistent=len(issues) == 0,
            internal_consistent=True,
            external_consistent=len(issues) == 0,
            inconsistencies=issues,
            suggestions=suggestions,
            adjustment=None,
        )
    
    def _check_compatibility(
        self,
        recommendation_type: str,
        equipment_type: str,
    ) -> str | None:
        """Check if recommendation is compatible with equipment."""
        # Incompatible combinations
        incompatible_map = {
            "sensor_replacement": {
                "defibrillator": "Perform defibrillator check",
                "infusion_pump": "Check infusion pump tubing",
            },
            "battery_replacement": {
                "monitor": "Check monitor power supply",
            },
        }
        
        alternatives = incompatible_map.get(recommendation_type, {}).get(equipment_type)
        return alternatives
    
    def _calculate_adjustment(
        self,
        internal_ok: bool,
        external_ok: bool,
        issue_count: int,
    ) -> float:
        """Calculate confidence adjustment based on consistency."""
        if internal_ok and external_ok:
            return 0.0  # No adjustment
        
        adjustment = 0.0
        
        if not internal_ok:
            adjustment -= 0.15  # Internal inconsistency penalty
        
        if not external_ok:
            adjustment -= 0.10  # External inconsistency penalty
        
        adjustment -= issue_count * 0.05  # Additional penalty per issue
        
        return max(-0.3, min(0.0, adjustment))  # Bounded to [-0.3, 0]
```

## Consecuencias

### Positivas
- ✅ Detecta inconsistencias internas
- ✅ Verifica consistencia con casos similares
- ✅ Ajusta confianza si es necesario

### Negativas
- ⚠️ Requiere acceso a casos históricos
