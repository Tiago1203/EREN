# ADR-3083: Self-Correction Design

**Título:** Diseño de Autocorrección
**Estado:** ✅ COMPLETO
**Fecha:** 2026-07-21
**Decisor:** Architecture Team

---

## Contexto

Capacidad de EREN para autocorrigirse cuando detecta errores o inconsistencias en sus recomendaciones.

---

## Decisión

### Sistema de Autocorrección

```python
class CorrectionType(Enum):
    """Types of self-corrections."""
    REWORD = "reword"                    # Rephrase recommendation
    NARROW = "narrow"                  # Make more specific
    BROADEN = "broaden"                # Make more general
    ADD_CONDITION = "add_condition"    # Add safety condition
    REDUCE_CONFIDENCE = "reduce"       # Reduce confidence level
    ESCALATE = "escalate"              # Request human input


@dataclass
class Correction:
    """Self-correction suggestion."""
    correction_id: str
    correction_type: CorrectionType
    original_text: str
    corrected_text: str
    reason: str
    confidence_impact: float


@dataclass
class CorrectionResult:
    """Result of self-correction."""
    was_corrected: bool
    corrections: list[Correction]
    final_recommendation: str | None
    requires_escalation: bool
    escalation_reason: str | None


class SelfCorrectionEngine:
    """Engine for self-correction of recommendations."""
    
    async def correct(
        self,
        recommendation: Recommendation,
        consistency_result: ConsistencyResult,
        evidence_result: EvidenceValidationResult,
    ) -> CorrectionResult:
        """Attempt to self-correct recommendation."""
        
        corrections = []
        reasons = []
        
        # Collect all issues
        reasons.extend(consistency_result.inconsistencies)
        if not evidence_result.is_sufficient:
            reasons.append(evidence_result.reason)
        
        if not reasons:
            return CorrectionResult(
                was_corrected=False,
                corrections=[],
                final_recommendation=recommendation.action,
                requires_escalation=False,
                escalation_reason=None,
            )
        
        # Determine correction type
        correction_type = self._determine_correction_type(reasons)
        
        if correction_type == CorrectionType.ESCALATE:
            return CorrectionResult(
                was_corrected=True,
                corrections=[],
                final_recommendation=None,
                requires_escalation=True,
                escalation_reason="; ".join(reasons),
            )
        
        # Apply correction
        correction = self._apply_correction(
            recommendation,
            correction_type,
            reasons,
        )
        corrections.append(correction)
        
        return CorrectionResult(
            was_corrected=True,
            corrections=corrections,
            final_recommendation=correction.corrected_text,
            requires_escalation=False,
            escalation_reason=None,
        )
    
    def _determine_correction_type(
        self,
        reasons: list[str],
    ) -> CorrectionType:
        """Determine what type of correction is needed."""
        
        for reason in reasons:
            # Too vague
            if "specificity" in reason.lower() or "vague" in reason.lower():
                return CorrectionType.NARROW
            
            # Needs safety condition
            if "safety" in reason.lower() or "risk" in reason.lower():
                return CorrectionType.ADD_CONDITION
            
            # Evidence quality issue
            if "quality" in reason.lower() or "evidence" in reason.lower():
                return CorrectionType.REDUCE_CONFIDENCE
            
            # Consistency issue
            if "consistent" in reason.lower():
                return CorrectionType.REWORD
        
        # Default: reduce confidence
        return CorrectionType.REDUCE_CONFIDENCE
    
    def _apply_correction(
        self,
        recommendation: Recommendation,
        correction_type: CorrectionType,
        reasons: list[str],
    ) -> Correction:
        """Apply specific correction type."""
        
        original = recommendation.action
        corrected = original
        confidence_impact = 0.0
        
        if correction_type == CorrectionType.REWORD:
            # Add clarifying language
            corrected = f"Consider: {original}"
            confidence_impact = -0.05
        
        elif correction_type == CorrectionType.NARROW:
            # Make more specific
            if "sensor" in original.lower():
                corrected = f"{original} (specific sensor model required)"
            else:
                corrected = f"{original} (review specifications)"
            confidence_impact = 0.0  # Neutral
        
        elif correction_type == CorrectionType.ADD_CONDITION:
            # Add safety condition
            corrected = f"{original} - ONLY if safety checks pass"
            confidence_impact = -0.10
        
        elif correction_type == CorrectionType.REDUCE_CONFIDENCE:
            # Reduce confidence level
            confidence_impact = -0.20
            corrected = f"{original} (verify with additional testing)"
        
        elif correction_type == CorrectionType.ESCALATE:
            corrected = "ESCALATE_TO_HUMAN"
            confidence_impact = -1.0
        
        return Correction(
            correction_id=f"corr_{datetime.now().timestamp()}",
            correction_type=correction_type,
            original_text=original,
            corrected_text=corrected,
            reason="; ".join(reasons),
            confidence_impact=confidence_impact,
        )
```

### Ejemplos de Autocorrección

```python
# Example 1: Vague recommendation
original = "Replace the part"
# Auto-corrected to:
corrected = "Replace the SpO2 sensor (specific sensor model required)"

# Example 2: Missing safety condition
original = "Bypass the alarm"
# Auto-corrected to:
corrected = "ESCALATE_TO_HUMAN"
# Reason: Dangerous action, cannot self-correct

# Example 3: Low confidence
original = "Recalibrate the monitor"
# Auto-corrected to:
corrected = "Recalibrate the monitor (verify with additional testing)"
# Confidence impact: -0.20
```

---

## Consecuencias

### Positivas
- ✅ EREN puede autocorregirse
- ✅ Reduce errores
- ✅ Añade condiciones de seguridad

### Negativas
- ⚠️ No todos los errores son corregibles
- ⚠️ Puede reducir confianza significativamente
