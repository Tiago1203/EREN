## Contexto

El cálculo de confianza combina múltiples fuentes de información para generar una puntuación única de 0-100%.

## Decisión

### Estrategias de Cálculo

```python
class EvidenceBasedCalculator:
    """Calculates confidence based on evidence."""
    
    async def calculate(self, evidence_bundle: EvidenceBundle) -> float:
        """Calculate evidence-based confidence."""
        
        # Evidence quantity
        evidence_count = len(evidence_bundle.supporting_evidence)
        quantity_score = min(evidence_count / 5, 1.0)  # Max at 5 evidence
        
        # Evidence quality (from sources)
        quality_scores = [e.quality_score for e in evidence_bundle.supporting_evidence]
        avg_quality = sum(quality_scores) / max(len(quality_scores), 1)
        
        # Evidence relevance
        relevance_scores = [e.relevance_score for e in evidence_bundle.supporting_evidence]
        avg_relevance = sum(relevance_scores) / max(len(relevance_scores), 1)
        
        # Contradicting evidence penalty
        contradiction_penalty = len(evidence_bundle.contradicting_evidence) * 0.1
        
        # Source diversity bonus
        sources = set(e.source_type for e in evidence_bundle.supporting_evidence)
        diversity_bonus = min(len(sources) / 3, 0.1)  # Max 0.1 bonus
        
        # Calculate final score
        score = (
            quantity_score * 0.3 +
            avg_quality * 0.3 +
            avg_relevance * 0.3 +
            diversity_bonus -
            contradiction_penalty
        )
        
        return max(0.0, min(1.0, score))


class KnowledgeBasedCalculator:
    """Calculates confidence based on knowledge."""
    
    async def calculate(self, context: dict) -> float:
        """Calculate knowledge-based confidence."""
        
        # Knowledge coverage
        coverage = context.get("knowledge_coverage", 0.0)
        
        # Knowledge quality
        quality = context.get("knowledge_quality", 0.0)
        
        # Knowledge recency
        recency = context.get("knowledge_recency", 0.5)
        
        # Domain expertise level
        expertise = context.get("domain_expertise", 0.5)
        
        score = (
            coverage * 0.3 +
            quality * 0.3 +
            recency * 0.2 +
            expertise * 0.2
        )
        
        return max(0.0, min(1.0, score))


class ReasoningBasedCalculator:
    """Calculates confidence based on reasoning."""
    
    async def calculate(self, chain: ReasoningChain) -> float:
        """Calculate reasoning-based confidence."""
        
        # Chain strength
        strength = chain.strength if hasattr(chain, 'strength') else 0.5
        
        # Step count (longer chains may be weaker)
        step_count = len(chain.steps) if hasattr(chain, 'steps') else 1
        step_penalty = max(0, (step_count - 5) * 0.02)
        
        # Premise support
        premises_supported = sum(1 for s in chain.steps if s.supported)
        premise_ratio = premises_supported / max(len(chain.steps), 1)
        
        # Conclusion consistency
        consistency = 0.9  # Placeholder
        
        score = (
            strength * 0.4 +
            premise_ratio * 0.3 +
            consistency * 0.3 -
            step_penalty
        )
        
        return max(0.0, min(1.0, score))


class EnsembleCalculator:
    """Combines multiple confidence sources."""
    
    def combine(
        self,
        evidence: float,
        knowledge: float,
        reasoning: float,
        weights: dict[str, float] | None = None,
    ) -> float:
        """Combine scores using weighted average."""
        
        if weights is None:
            weights = {"evidence": 0.4, "knowledge": 0.3, "reasoning": 0.3}
        
        total_score = (
            evidence * weights.get("evidence", 0.4) +
            knowledge * weights.get("knowledge", 0.3) +
            reasoning * weights.get("reasoning", 0.3)
        )
        
        return max(0.0, min(1.0, total_score))
    
    def combine_uncertain(
        self,
        scores: list[float],
        uncertainties: list[float],
    ) -> tuple[float, float]:
        """Combine scores with uncertainty propagation."""
        
        if not scores:
            return 0.5, 1.0
        
        # Weighted average
        total_score = sum(scores) / len(scores)
        
        # Propagate uncertainty
        # Larger spread = more uncertainty
        import statistics
        spread = statistics.stdev(scores) if len(scores) > 1 else 0
        
        # Combined uncertainty
        avg_uncertainty = sum(uncertainties) / len(uncertainties)
        combined_uncertainty = avg_uncertainty + spread * 0.5
        
        return total_score, min(combined_uncertainty, 1.0)
```

# Alta confianza (92%)
evidence_score = 0.95  # 5 strong evidence items

reasoning_score = 0.92  # Strong chain

combined = ensemble.combine(evidence_score, knowledge_score, reasoning_score)

# Result: 0.31 (31%)
