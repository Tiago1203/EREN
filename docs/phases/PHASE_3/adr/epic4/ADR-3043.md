## Contexto

La cuantificación de incertidumbre identifica y mide los factores que reducen la confianza en las recomendaciones de EREN.

---

## Decisión

### Modelo de Incertidumbre

```python
class UncertaintyType(Enum):
    """Types of uncertainty."""
    EPISTEMIC = "epistemic"       # Lack of knowledge
    ALEATORY = "aleatory"         # Inherent randomness
    MEASUREMENT = "measurement"   # Measurement error
    MODEL = "model"              # Model limitations
    CONTEXT = "context"          # Context dependence


@dataclass(frozen=True)
class UncertaintyFactor:
    """Individual uncertainty factor."""
    factor_id: str
    type: UncertaintyType
    description: str
    magnitude: float              # 0.0 - 1.0
    impact_on_confidence: float   # How much it reduces confidence
    mitigatable: bool             # Can be reduced with more info


@dataclass(frozen=True)
class UncertaintyReport:
    """Complete uncertainty quantification."""
    total_uncertainty: float     # Overall uncertainty (0.0 - 1.0)
    uncertainty_type: UncertaintyType
    factors: list[UncertaintyFactor]
    confidence_impact: float      # How much confidence is reduced
    mitigation_suggestions: list[str]
    data_gaps: list[str]


class UncertaintyQuantifier:
    """Quantifies uncertainty in confidence assessment."""
    
    async def quantify(
        self,
        hypothesis: Hypothesis,
        evidence_bundle: EvidenceBundle,
        context: dict,
    ) -> UncertaintyReport:
        """Quantify uncertainty for a hypothesis."""
        
        factors = []
        
        # Epistemic uncertainty (lack of knowledge)
        if context.get("knowledge_coverage", 1.0) < 0.5:
            factors.append(UncertaintyFactor(
                factor_id="epistemic_knowledge",
                type=UncertaintyType.EPISTEMIC,
                description="Limited knowledge coverage in domain",
                magnitude=1.0 - context.get("knowledge_coverage", 0.5),
                impact_on_confidence=0.2,
                mitigatable=True,
            ))
        
        # Evidence uncertainty
        evidence_count = len(evidence_bundle.supporting_evidence)
        if evidence_count < 3:
            factors.append(UncertaintyFactor(
                factor_id="evidence_scarcity",
                type=UncertaintyType.EPISTEMIC,
                description=f"Limited evidence ({evidence_count} items)",
                magnitude=1.0 - (evidence_count / 5),
                impact_on_confidence=0.25,
                mitigatable=True,
            ))
        
        # Conflicting evidence
        if evidence_bundle.contradicting_evidence:
            contradiction_ratio = (
                len(evidence_bundle.contradicting_evidence) /
                (len(evidence_bundle.supporting_evidence) + 1)
            )
            factors.append(UncertaintyFactor(
                factor_id="contradicting_evidence",
                type=UncertaintyType.EPISTEMIC,
                description="Evidence contradicts hypothesis",
                magnitude=contradiction_ratio,
                impact_on_confidence=contradiction_ratio * 0.3,
                mitigatable=True,
            ))
        
        # Measurement uncertainty
        if context.get("has_measurements", False):
            measurement_precision = context.get("measurement_precision", 1.0)
            if measurement_precision < 0.8:
                factors.append(UncertaintyFactor(
                    factor_id="measurement_precision",
                    type=UncertaintyType.MEASUREMENT,
                    description="Low measurement precision",
                    magnitude=1.0 - measurement_precision,
                    impact_on_confidence=0.15,
                    mitigatable=True,
                ))
        
        # Model uncertainty
        if context.get("uses_heuristics", False):
            factors.append(UncertaintyFactor(
                factor_id="model_heuristics",
                type=UncertaintyType.MODEL,
                description="Using heuristic approximations",
                magnitude=0.3,
                impact_on_confidence=0.1,
                mitigatable=False,
            ))
        
        # Calculate total uncertainty
        total = self._calculate_total_uncertainty(factors)
        
        # Generate mitigation suggestions
        mitigations = self._generate_mitigations(factors)
        
        # Identify data gaps
        gaps = self._identify_data_gaps(evidence_bundle, context)
        
        return UncertaintyReport(
            total_uncertainty=total,
            uncertainty_type=self._dominant_type(factors),
            factors=factors,
            confidence_impact=sum(f.impact_on_confidence for f in factors),
            mitigation_suggestions=mitigations,
            data_gaps=gaps,
        )
    
    def _calculate_total_uncertainty(self, factors: list[UncertaintyFactor]) -> float:
        """Calculate total uncertainty using aggregation."""
        if not factors:
            return 0.0
        
        # Use max for independent uncertainties
        max_magnitude = max(f.magnitude for f in factors)
        
        # Add complementary uncertainties
        complementary = [
            f.magnitude for f in factors 
            if f.type in [UncertaintyType.EPISTEMIC, UncertaintyType.MEASUREMENT]
        ]
        
        if complementary:
            # Sum of complementary (bounded)
            sum_uncertainty = min(sum(complementary), 1.0)
            return max(max_magnitude, sum_uncertainty * 0.5)
        
        return max_magnitude
    
    def _dominant_type(self, factors: list[UncertaintyFactor]) -> UncertaintyType:
        """Find dominant uncertainty type."""
        if not factors:
            return UncertaintyType.EPISTEMIC
        
        type_scores: dict[UncertaintyType, float] = {}
        for f in factors:
            type_scores[f.type] = type_scores.get(f.type, 0) + f.magnitude
        
        return max(type_scores, key=type_scores.get)
    
    def _generate_mitigations(self, factors: list[UncertaintyFactor]) -> list[str]:
        """Generate suggestions to reduce uncertainty."""
        mitigations = []
        
        for f in factors:
            if f.mitigatable:
                if f.type == UncertaintyType.EPISTEMIC:
                    if "evidence" in f.factor_id:
                        mitigations.append("Gather more evidence from authoritative sources")
                    elif "knowledge" in f.factor_id:
                        mitigations.append("Consult additional knowledge sources")
                elif f.type == UncertaintyType.MEASUREMENT:
                    mitigations.append("Improve measurement precision")
        
        return list(set(mitigations))
    
    def _identify_data_gaps(
        self,
        evidence_bundle: EvidenceBundle,
        context: dict,
    ) -> list[str]:
        """Identify specific data gaps."""
        gaps = []
        
        if len(evidence_bundle.supporting_evidence) < 3:
            gaps.append("Insufficient supporting evidence")
        
        if not evidence_bundle.contradicting_evidence:
            gaps.append("No contradicting evidence considered")
        
        if context.get("knowledge_coverage", 1.0) < 0.7:
            gaps.append("Limited domain knowledge coverage")
        
        if not context.get("has_historical_data", False):
            gaps.append("No historical data available")
        
        return gaps
```
