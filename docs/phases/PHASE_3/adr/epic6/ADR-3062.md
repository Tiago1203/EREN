### Alarm Rules

```python
class AlarmRules:
    """Clinical alarm safety rules."""
    
    RULES = [
        {
            "id": "ALARM_DELAY_RESP",
            "name": "Respiratory Alarm Delay",
            "description": "Respiratory alarm must activate within 30 seconds",
            "condition": "alarm_delay_seconds <= 30",
            "severity": "HIGH",
            "source": "AAMI Alarm Safety Guidelines",
        },
        {
            "id": "ALARM_DELAY_HR",
            "name": "Heart Rate Alarm Delay",
            "description": "HR alarm must activate within 30 seconds",
            "condition": "alarm_delay_seconds <= 30",
            "severity": "HIGH",
            "source": "AAMI Alarm Safety Guidelines",
        },
        {
            "id": "ALARM_SATURATION",
            "name": "SpO2 Low Alarm",
            "description": "Low SpO2 alarm must be set appropriately",
            "condition": "spo2_low_alarm <= patient_baseline - 5",
            "severity": "CRITICAL",
            "source": "AAMI Alarm Safety Guidelines",
        },
        {
            "id": "ALARM_SILENCE_MAX",
            "name": "Maximum Alarm Silence Duration",
            "description": "Alarm silence must not exceed 2 minutes",
            "condition": "alarm_silence_duration_minutes <= 2",
            "severity": "HIGH",
            "source": "AAMI Alarm Safety Guidelines",
        },
    ]


class AlarmValidator:
    """Validates alarm configuration."""
    
    async def validate(self, equipment: Equipment, context: dict) -> list[Violation]:
        """Validate alarm settings."""
        violations = []
        
        for rule in AlarmRules.RULES:
            if not self._evaluate_condition(rule, equipment, context):
                violations.append(Violation(
                    rule_id=rule["id"],
                    description=rule["description"],
                    severity=rule["severity"],
                    recommendation=f"Adjust {rule['name']}",
                ))
        
        return violations
```

### Safety Rules

```python
class ClinicalSafetyRules:
    """Clinical safety rules."""
    
    RULES = [
        {
            "id": "SAFETY_PATIENT_ISOLATION",
            "name": "Patient Isolation Required",
            "description": "Isolated patients require isolated equipment",
            "condition": "equipment.isolation_compliant == True",
            "severity": "CRITICAL",
            "action": "IMMEDIATE - Use isolated equipment",
        },
        {
            "id": "SAFETY_EMERGENCY_STOP",
            "name": "Emergency Stop Accessible",
            "description": "Emergency stop must be within reach",
            "condition": "emergency_stop_accessible == True",
            "severity": "CRITICAL",
            "action": "Ensure emergency stop is accessible",
        },
        {
            "id": "SAFETY_LEAKAGE_CURRENT",
            "name": "Leakage Current Check",
            "description": "Leakage current must be within safe limits",
            "condition": "leakage_current_ma <= 0.5",
            "severity": "CRITICAL",
            "action": "IMMEDIATE REMOVAL - Unsafe leakage current",
        },
        {
            "id": "SAFETY_GROUNDING",
            "name": "Proper Grounding",
            "description": "Equipment must be properly grounded",
            "condition": "ground_resistance_ohm <= 0.2",
            "severity": "CRITICAL",
            "action": "IMMEDIATE - Fix grounding issue",
        },
    ]


class SafetyValidator:
    """Validates clinical safety requirements."""
    
    async def validate(self, equipment: Equipment, context: dict) -> list[Violation]:
        """Validate safety requirements."""
        violations = []
        
        for rule in ClinicalSafetyRules.RULES:
            if not self._evaluate_condition(rule, equipment, context):
                violations.append(Violation(
                    rule_id=rule["id"],
                    description=rule["description"],
                    severity=rule["severity"],
                    action=rule.get("action"),
                ))
        
        return violations
```
