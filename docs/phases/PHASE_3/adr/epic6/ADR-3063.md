### Maintenance Rules

```python
class MaintenanceRules:
    """Equipment maintenance rules."""
    
    # Equipment-specific intervals (days)
    EQUIPMENT_INTERVALS = {
        "Ventilator": {
            "preventive_maintenance": 180,
            "filter_replacement": 90,
            "sensor_calibration": 365,
        },
        "Infusion Pump": {
            "preventive_maintenance": 365,
            "battery_replacement": 730,
            "accuracy_check": 180,
        },
        "Defibrillator": {
            "preventive_maintenance": 180,
            "battery_check": 30,
            "electrode_expiry_check": 90,
        },
        "SpO2 Monitor": {
            "preventive_maintenance": 365,
            "sensor_calibration": 180,
            "accuracy_verification": 90,
        },
        "ECG Monitor": {
            "preventive_maintenance": 365,
            "electrode_check": 180,
            "leakage_test": 180,
        },
        "Infant Incubator": {
            "preventive_maintenance": 180,
            "temperature_calibration": 90,
            "humidity_check": 30,
        },
    }
    
    RULES = [
        {
            "id": "MAINT_PM_OVERDUE",
            "name": "Preventive Maintenance Overdue",
            "description": "Equipment PM is overdue",
            "condition": "days_since_pm <= equipment_interval",
            "severity": "HIGH",
            "action": "Schedule preventive maintenance",
        },
        {
            "id": "MAINT_FILTER_REPLACE",
            "name": "Filter Replacement Due",
            "description": "Equipment filter should be replaced",
            "condition": "days_since_filter_replacement <= filter_interval",
            "severity": "MEDIUM",
            "action": "Replace filter during next PM",
        },
    ]


class MaintenanceValidator:
    """Validates maintenance requirements."""
    
    async def validate(self, equipment: Equipment, context: dict) -> list[Violation]:
        """Validate maintenance status."""
        violations = []
        
        equipment_type = context.get("equipment_type")
        intervals = MaintenanceRules.EQUIPMENT_INTERVALS.get(equipment_type, {})
        
        # Check PM interval
        pm_interval = intervals.get("preventive_maintenance", 365)
        days_since_pm = context.get("days_since_pm", 0)
        
        if days_since_pm > pm_interval:
            violations.append(Violation(
                rule_id="MAINT_PM_OVERDUE",
                description=f"PM overdue by {days_since_pm - pm_interval} days",
                severity="HIGH",
                action=f"Schedule PM (interval: {pm_interval} days)",
                equipment_type=equipment_type,
            ))
        
        return violations
```

### Calibration Rules

```python
class CalibrationRules:
    """Equipment calibration rules."""
    
    RULES = [
        {
            "id": "CALIBRATION_OVERDUE",
            "name": "Calibration Overdue",
            "description": "Equipment calibration is overdue",
            "condition": "days_since_calibration <= calibration_interval",
            "severity": "HIGH",
            "action": "Schedule calibration",
        },
        {
            "id": "CALIBRATION_TRACEABILITY",
            "name": "Calibration Traceability",
            "description": "Calibration must be traceable to NIST",
            "condition": "calibration_certificate.traceable == True",
            "severity": "HIGH",
            "action": "Verify calibration traceability",
        },
        {
            "id": "CALIBRATION_ACCURACY",
            "name": "Calibration Accuracy",
            "description": "Equipment must meet accuracy specifications",
            "condition": "measurement_error <= acceptable_error",
            "severity": "CRITICAL",
            "action": "Adjust or replace equipment",
        },
    ]


class CalibrationValidator:
    """Validates calibration requirements."""
    
    async def validate(self, equipment: Equipment, context: dict) -> list[Violation]:
        """Validate calibration status."""
        violations = []
        
        calibration_interval = context.get("calibration_interval_days", 365)
        days_since_calibration = context.get("days_since_calibration", 0)
        
        if days_since_calibration > calibration_interval:
            overdue_days = days_since_calibration - calibration_interval
            violations.append(Violation(
                rule_id="CALIBRATION_OVERDUE",
                description=f"Calibration overdue by {overdue_days} days",
                severity="HIGH",
                action="Schedule calibration immediately",
            ))
        
        return violations
```

### Electrical Rules

```python
class ElectricalRules:
    """Electrical safety rules."""
    
    RULES = [
        {
            "id": "ELECTRICAL_LEAKAGE",
            "name": "Leakage Current Check",
            "description": "Leakage current must be within limits",
            "limit_ma": 0.5,
            "severity": "CRITICAL",
        },
        {
            "id": "ELECTRICAL_GROUND",
            "name": "Ground Resistance",
            "description": "Ground resistance must be low",
            "limit_ohm": 0.2,
            "severity": "CRITICAL",
        },
        {
            "id": "ELECTRICAL_BATTERY",
            "name": "Battery Capacity",
            "description": "Battery must maintain adequate capacity",
            "minimum_capacity_percent": 80,
            "severity": "HIGH",
        },
        {
            "id": "ELECTRICAL_BATTERY_CYCLES",
            "name": "Battery Cycle Count",
            "description": "Battery cycle count must not exceed limit",
            "maximum_cycles": 500,
            "severity": "HIGH",
        },
    ]


class ElectricalValidator:
    """Validates electrical safety."""
    
    async def validate(self, equipment: Equipment, context: dict) -> list[Violation]:
        """Validate electrical requirements."""
        violations = []
        
        # Check leakage current
        leakage_current = context.get("leakage_current_ma", 0)
        if leakage_current > 0.5:
            violations.append(Violation(
                rule_id="ELECTRICAL_LEAKAGE",
                description=f"Leakage current {leakage_current}mA exceeds 0.5mA limit",
                severity="CRITICAL",
                action="IMMEDIATE REMOVAL FROM SERVICE",
            ))
        
        # Check battery cycles
        battery_cycles = context.get("battery_cycles", 0)
        if battery_cycles > 500:
            violations.append(Violation(
                rule_id="ELECTRICAL_BATTERY_CYCLES",
                description=f"Battery cycles ({battery_cycles}) exceeds 500 limit",
                severity="HIGH",
                action="Replace battery",
            ))
        
        return violations
```
