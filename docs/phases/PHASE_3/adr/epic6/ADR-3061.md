## Contexto

Validación contra estándares IEC para equipos médicos.

## Decisión

### Estándares IEC Cubiertos

```
IEC 60601-1      - Requisitos generales para seguridad
IEC 60601-2-61   - Requisitos particulares para oxímetros de pulso
IEC 62304        - Ciclo de vida del software de dispositivos médicos
IEC 62353        - Pruebas recurrentes de equipos médicos
```

### Reglas IEC 60601-1

```python
class IEC60601Validator:
    """Validator for IEC 60601-1."""
    
    RULES = [
        {
            "id": "IEC60601_1_LEAKAGE_CURRENT",
            "clause": "8.7",
            "name": "Leakage Current Limits",
            "description": "Equipment must not exceed leakage current limits",
            "rule": {
                "condition": "leakage_current_ma <= 0.5",
                "unit": "mA",
                "measurement_point": "Patient connection",
            },
            "severity": "CRITICAL",
            "action": "IMMEDIATE REMOVAL FROM SERVICE",
        },
        {
            "id": "IEC60601_1_GROUNDING",
            "clause": "8.6",
            "name": "Protective Grounding",
            "description": "Equipment must have proper protective grounding",
            "rule": {
                "condition": "ground_resistance <= 0.2",
                "unit": "ohm",
            },
            "severity": "CRITICAL",
            "action": "IMMEDIATE REMOVAL FROM SERVICE",
        },
        {
            "id": "IEC60601_1_DIELECTRIC",
            "clause": "8.8",
            "name": "Dielectric Strength",
            "description": "Isolation must withstand test voltage",
            "rule": {
                "condition": "dielectric_withstand == PASS",
            },
            "severity": "HIGH",
            "action": "Remove from service pending retest",
        },
        {
            "id": "IEC60601_1_CALIBRATION",
            "clause": "4.4",
            "name": "Calibration Interval",
            "description": "Equipment must be calibrated per manufacturer specs",
            "rule": {
                "condition": "days_since_calibration <= calibration_interval",
                "typical_interval": 365,  # days
            },
            "severity": "MEDIUM",
            "action": "Schedule calibration",
        },
        {
            "id": "IEC60601_1_MAINTENANCE",
            "clause": "5.1",
            "name": "Preventive Maintenance",
            "description": "Equipment must receive preventive maintenance",
            "rule": {
                "condition": "days_since_pm <= pm_interval",
                "ventilator_interval": 180,  # days
            },
            "severity": "HIGH",
            "action": "Schedule preventive maintenance",
        },
    ]
    
    async def validate(self, equipment, context) -> ValidationResult:
        """Validate against IEC 60601-1."""
        violations = []
        
        for rule in self.RULES:
            result = await self._check_rule(rule, equipment, context)
            if not result.passed:
                violations.append(Violation(
                    rule_id=rule["id"],
                    standard="IEC 60601-1",
                    clause=rule["clause"],
                    description=rule["description"],
                    severity=rule["severity"],
                    current_value=result.current_value,
                    required_value=rule["rule"].get("condition"),
                ))
        
        return ValidationResult(
            is_compliant=len(violations) == 0,
            violations=violations,
            standard="IEC 60601-1",
        )
```

class IEC60601_2_61Validator:
