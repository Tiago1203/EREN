## Decisión

### Pesos de Componentes

```
├── Evidence Score: 40%
│   └── Cantidad, calidad, relevancia de evidencia
├── Confidence Score: 25%
│   └── Confianza del reasoning
├── Safety Score: 20%
│   └── Nivel de riesgo de seguridad
└── Rules Score: 15%
    └── Cumplimiento de reglas IEC/FDA
```

### Algoritmo de Puntuación

```python
class DecisionScorer:
    """Scores decisions based on multiple factors."""
    
    # Component weights
    EVIDENCE_WEIGHT = 0.40
    CONFIDENCE_WEIGHT = 0.25
    SAFETY_WEIGHT = 0.20
    RULES_WEIGHT = 0.15
    
    def score(
        self,
        alternative: Alternative,
        context: dict,
        safety_result: SafetyResult,
        rules_result: ValidationResult,
        confidence_score: float,
    ) -> DecisionScore:
        """Calculate total score for alternative."""
        
        # Evidence score (0-1)
        evidence_bundle = context.get("evidence_bundle", {})
        evidence_score = self._calculate_evidence_score(evidence_bundle)
        
        # Safety score (0-1)
        safety_score = 1.0 if safety_result.is_safe else 0.0
        
        # Rules score (0-1)
        rules_score = 1.0 if rules_result.is_compliant else 0.0
        
        # Confidence score (already 0-1)
        conf_score = confidence_score
        
        # Weighted sum
        total_score = (
            evidence_score * self.EVIDENCE_WEIGHT +
            conf_score * self.CONFIDENCE_WEIGHT +
            safety_score * self.SAFETY_WEIGHT +
            rules_score * self.RULES_WEIGHT
        )
        
        return DecisionScore(
            total_score=total_score,
            evidence_score=evidence_score,
            confidence_score=conf_score,
            safety_score=safety_score,
            rules_score=rules_score,
        )
    
    def _calculate_evidence_score(
        self,
        evidence_bundle: EvidenceBundle,
    ) -> float:
        """Calculate evidence score."""
        supporting = evidence_bundle.get("supporting", [])
        
        if not supporting:
            return 0.0
        
        # Count factor
        count_score = min(len(supporting) / 3, 1.0)  # 3 = ideal
        
        # Quality factor
        quality_scores = [e.get("quality_score", 0) for e in supporting]
        quality_score = sum(quality_scores) / len(quality_scores) if quality_scores else 0
        
        # Relevance factor
        relevance_scores = [e.get("relevance_score", 0) for e in supporting]
        relevance_score = sum(relevance_scores) / len(relevance_scores) if relevance_scores else 0
        
        # Combined
        return (count_score * 0.3 + quality_score * 0.4 + relevance_score * 0.3)


class AlternativeRanker:
    """Ranks decision alternatives."""
    
    def rank(
        self,
        scored_alternatives: list[tuple[Alternative, DecisionScore]],
    ) -> list[DecisionAlternative]:
        """Rank alternatives by score."""
        
        # Sort by total score descending
        sorted_alts = sorted(
            scored_alternatives,
            key=lambda x: x[1].total_score,
            reverse=True
        )
        
        # Assign rankings
        ranked = []
        for rank, (alt, score) in enumerate(sorted_alts, 1):
            ranked.append(DecisionAlternative(
                id=alt.id,
                option=alt.option,
                score=score.total_score,
                evidence_count=len(alt.evidence.get("supporting", [])),
                evidence_bundle=alt.evidence,
                risk_level=alt.risk_level,
                confidence=alt.confidence,
                ranking=rank,
            ))
        
        return ranked
```
