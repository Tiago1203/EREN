### Modelo Principal

```python
@dataclass
class ClinicalDecision:
    """Main decision object."""
    id: str
    timestamp: datetime
    decision: str
    confidence: float
    severity: Severity
    priority: Priority
    evidence: EvidenceBundle
    reasoning: ReasoningChain
    risks: list[Risk]
    alternatives: list[DecisionAlternative]
    recommended_actions: list[DecisionAction]
    validation_status: ValidationStatus
    automation_level: AutomationLevel
    audit: AuditInfo


class DecisionEngine:
    """Main decision engine."""
    
    def __init__(self):
        self.ranker = AlternativeRanker()
        self.scorer = DecisionScorer()
        self.planner = ActionPlanner()
        self.priority_classifier = PriorityClassifier()
        self.automation_evaluator = AutomationEvaluator()
        self.recommendation_generator = RecommendationGenerator()
        self.recorder = DecisionRecorder()
    
    async def decide(
        self,
        alternatives: list[Alternative],
        context: dict,
        safety_result: SafetyResult,
        rules_result: ValidationResult,
        confidence_score: ConfidenceScore,
    ) -> ClinicalDecision:
        """Make the final clinical decision."""
        
        # 1. Score all alternatives
        scored = await self._score_alternatives(
            alternatives, context
        )
        
        # 2. Rank alternatives
        ranked = self.ranker.rank(scored)
        
        # 3. Select best alternative
        best = ranked[0] if ranked else None
        
        if not best:
            return ClinicalDecision(
                id=self._generate_id(),
                decision="NO_DECISION",
                confidence=0.0,
                # ... other fields
            )
        
        # 4. Generate action plan
        action_plan = await self.planner.create_plan(
            best.option, context
        )
        
        # 5. Classify priority
        priority = self.priority_classifier.classify(
            best, context
        )
        
        # 6. Evaluate automation level
        automation_level = self.automation_evaluator.evaluate(
            best, safety_result, rules_result, confidence_score
        )
        
        # 7. Generate recommendations
        recommendations = self.recommendation_generator.generate(
            best, action_plan
        )
        
        # 8. Create decision
        decision = ClinicalDecision(
            id=self._generate_id(),
            timestamp=datetime.now(),
            decision=best.option,
            confidence=best.confidence,
            priority=priority,
            automation_level=automation_level,
            action_plan=action_plan,
            recommendations=recommendations,
            # ... other fields
        )
        
        # 9. Record for audit
        await self.recorder.record(decision)
        
        return decision
```
