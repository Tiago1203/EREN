class ActionPlanner:
    """Generates action plans for decisions."""
    
    def create_plan(
        self,
        decision: str,
        context: dict,
    ) -> DecisionPlan:
        """Create action plan for decision."""
        
        # Determine plan template
        plan_type = self._classify_plan_type(decision)
        
        # Generate steps
        steps = self._generate_steps(plan_type, context)
        
        # Calculate total duration
        total_duration = sum(
            s.estimated_duration for s in steps
        )
        
        # Determine prerequisites
        prerequisites = self._get_prerequisites(plan_type)
        
        # Add safety checks
        safety_checks = self._get_safety_checks(plan_type)
        
        return DecisionPlan(
            steps=steps,
            estimated_total_duration=total_duration,
            prerequisites=prerequisites,
            safety_checks=safety_checks,
        )
    
    def _classify_plan_type(self, decision: str) -> str:
        """Classify the type of action plan needed."""
        decision_lower = decision.lower()
        
        if "replace" in decision_lower:
            return "REPLACEMENT"
        elif "calibrat" in decision_lower:
            return "CALIBRATION"
        elif "repair" in decision_lower:
            return "REPAIR"
        elif "maintenance" in decision_lower:
            return "MAINTENANCE"
        elif "update" in decision_lower or "firmware" in decision_lower:
            return "UPDATE"
        else:
            return "GENERAL"

@dataclass
class PlanStep:
    """Individual step in action plan."""
    step_number: int
    action: str
    description: str
    estimated_duration: timedelta
    required_personnel: list[str]
    safety_warnings: list[str]


def _generate_steps(self, plan_type: str, context: dict) -> list[PlanStep]:
    """Generate steps from template."""
    template = PLAN_TEMPLATES.get(plan_type, PLAN_TEMPLATES["GENERAL"])
    
    steps = []
    personnel = self._determine_required_personnel(plan_type)
    
    for i, step_template in enumerate(template["steps"], 1):
        step = PlanStep(
            step_number=i,
            action=step_template["action"],
            description=f"Step {i}: {step_template['action']}",
            estimated_duration=timedelta(minutes=step_template["duration_min"]),
            required_personnel=personnel,
            safety_warnings=template["safety_checks"],
        )
        steps.append(step)
    
    return steps
