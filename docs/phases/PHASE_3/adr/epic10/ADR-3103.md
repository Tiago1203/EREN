## Decisión

### Pattern Types

```python
class PatternType(Enum):
    """Types of discovered patterns."""
    TIME_BASED = "time_based"              # Seasonal, hourly patterns
    DEVICE_SPECIFIC = "device_specific"    # Model-specific failures
    ENVIRONMENTAL = "environmental"        # Temperature, humidity
    RECOMMENDATION = "recommendation"      # Successful vs failed recs
    SYMPTOM_DIAGNOSIS = "symptom_diagnosis"  # New mappings


@dataclass
class Pattern:
    """Discovered pattern."""
    pattern_id: str
    pattern_type: PatternType
    description: str
    confidence: float
    supporting_cases: list[str]
    statistical_significance: float
    discovered_at: datetime
    last_validated: datetime | None


@dataclass
class PatternDiscoveryResult:
    """Result of pattern discovery."""
    patterns: list[Pattern]
    high_confidence_patterns: list[Pattern]
    requires_human_review: list[Pattern]
    discovery_metrics: dict
```

### Pattern Discovery

```python
class PatternDiscovery:
    """Discovers patterns in experience data."""
    
    def __init__(self):
        self.min_supporting_cases = 5
        self.min_significance = 0.8
    
    async def discover(
        self,
        cases: list[ExperienceCase],
    ) -> PatternDiscoveryResult:
        """Discover patterns in cases."""
        
        patterns = []
        
        # Discover time patterns
        time_patterns = await self._discover_time_patterns(cases)
        patterns.extend(time_patterns)
        
        # Discover device patterns
        device_patterns = await self._discover_device_patterns(cases)
        patterns.extend(device_patterns)
        
        # Discover recommendation patterns
        rec_patterns = await self._discover_recommendation_patterns(cases)
        patterns.extend(rec_patterns)
        
        # Classify by confidence
        high_confidence = [p for p in patterns if p.confidence >= 0.9]
        requires_review = [p for p in patterns if p.confidence < 0.9]
        
        return PatternDiscoveryResult(
            patterns=patterns,
            high_confidence_patterns=high_confidence,
            requires_human_review=requires_review,
            discovery_metrics=self._calculate_metrics(patterns),
        )
    
    async def _discover_time_patterns(
        self,
        cases: list[ExperienceCase],
    ) -> list[Pattern]:
        """Discover time-based patterns."""
        patterns = []
        
        # Group by month
        monthly_outcomes = {}
        for case in cases:
            month = case.timestamp.month
            if month not in monthly_outcomes:
                monthly_outcomes[month] = {"success": 0, "failure": 0}
            
            if case.outcome == OutcomeType.SUCCESS:
                monthly_outcomes[month]["success"] += 1
            else:
                monthly_outcomes[month]["failure"] += 1
        
        # Find seasonal patterns
        for month, outcomes in monthly_outcomes.items():
            total = outcomes["success"] + outcomes["failure"]
            if total >= self.min_supporting_cases:
                failure_rate = outcomes["failure"] / total
                
                # High failure rate in specific month
                if failure_rate > 0.3:
                    patterns.append(Pattern(
                        pattern_id=f"time_pattern_month_{month}",
                        pattern_type=PatternType.TIME_BASED,
                        description=f"High failure rate in month {month}: {failure_rate:.0%}",
                        confidence=failure_rate,
                        supporting_cases=[],
                        statistical_significance=self._calculate_significance(total),
                        discovered_at=datetime.now(),
                        last_validated=None,
                    ))
        
        return patterns
    
    async def _discover_device_patterns(
        self,
        cases: list[ExperienceCase],
    ) -> list[Pattern]:
        """Discover device-specific patterns."""
        patterns = []
        
        # Group by device type
        device_outcomes = {}
        for case in cases:
            device_type = case.device_type
            if device_type not in device_outcomes:
                device_outcomes[device_type] = {"success": 0, "failure": 0}
            
            if case.outcome == OutcomeType.SUCCESS:
                device_outcomes[device_type]["success"] += 1
            else:
                device_outcomes[device_type]["failure"] += 1
        
        # Find device-specific patterns
        for device_type, outcomes in device_outcomes.items():
            total = outcomes["success"] + outcomes["failure"]
            if total >= self.min_supporting_cases:
                success_rate = outcomes["success"] / total
                
                patterns.append(Pattern(
                    pattern_id=f"device_pattern_{device_type}",
                    pattern_type=PatternType.DEVICE_SPECIFIC,
                    description=f"{device_type} success rate: {success_rate:.0%}",
                    confidence=success_rate,
                    supporting_cases=[],
                    statistical_significance=self._calculate_significance(total),
                    discovered_at=datetime.now(),
                    last_validated=None,
                ))
        
        return patterns
    
    async def _discover_recommendation_patterns(
        self,
        cases: list[ExperienceCase],
    ) -> list[Pattern]:
        """Discover recommendation success patterns."""
        patterns = []
        
        # Group by decision
        decision_outcomes = {}
        for case in cases:
            decision = case.decision_made
            if decision not in decision_outcomes:
                decision_outcomes[decision] = {"success": 0, "failure": 0}
            
            if case.outcome == OutcomeType.SUCCESS:
                decision_outcomes[decision]["success"] += 1
            else:
                decision_outcomes[decision]["failure"] += 1
        
        # Find recommendation patterns
        for decision, outcomes in decision_outcomes.items():
            total = outcomes["success"] + outcomes["failure"]
            if total >= self.min_supporting_cases:
                success_rate = outcomes["success"] / total
                
                patterns.append(Pattern(
                    pattern_id=f"rec_pattern_{hash(decision)}",
                    pattern_type=PatternType.RECOMMENDATION,
                    description=f"Recommendation '{decision[:50]}' success rate: {success_rate:.0%}",
                    confidence=success_rate,
                    supporting_cases=[],
                    statistical_significance=self._calculate_significance(total),
                    discovered_at=datetime.now(),
                    last_validated=None,
                ))
        
        return patterns
    
    def _calculate_significance(self, sample_size: int) -> float:
        """Calculate statistical significance based on sample size."""
        # Simple significance based on sample size
        if sample_size >= 20:
            return 0.95
        elif sample_size >= 10:
            return 0.85
        elif sample_size >= 5:
            return 0.70
        return 0.50
```
