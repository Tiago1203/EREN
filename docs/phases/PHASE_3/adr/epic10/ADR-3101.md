## Decisión

### Outcome Types

```python
class OutcomeType(Enum):
    """Types of decision outcomes."""
    SUCCESS = "success"                  # Problem solved
    PARTIAL_SUCCESS = "partial_success"  # Problem partially solved
    FAILURE = "failure"                  # Problem not solved
    NO_ACTION = "no_action"             # Recommendation not followed
    UNKNOWN = "unknown"                  # Outcome unclear


@dataclass
class Outcome:
    """Outcome of a clinical decision."""
    decision_id: str
    outcome_type: OutcomeType
    timestamp: datetime
    evidence: list[str]
    description: str
    resolution_time_minutes: int | None
    side_effects: list[str]


@dataclass
class CollectedOutcome:
    """Outcome after collection and verification."""
    outcome: Outcome
    verified: bool
    verification_evidence: list[str]
    similar_historical_outcomes: list[Outcome]
    recommended_confidence_adjustment: float
```

### Outcome Collector

```python
class OutcomeCollector:
    """Collects and verifies decision outcomes."""
    
    async def collect(
        self,
        decision: ClinicalDecision,
        outcome: Outcome,
    ) -> CollectedOutcome:
        """Collect and verify outcome."""
        
        # Verify outcome evidence
        verified = self._verify_outcome(outcome)
        
        # Find similar historical outcomes
        similar = await self._find_similar_outcomes(
            decision, outcome
        )
        
        # Calculate confidence adjustment
        adjustment = self._calculate_adjustment(
            outcome, similar
        )
        
        return CollectedOutcome(
            outcome=outcome,
            verified=verified,
            verification_evidence=self._get_evidence(outcome),
            similar_historical_outcomes=similar,
            recommended_confidence_adjustment=adjustment,
        )
    
    def _verify_outcome(self, outcome: Outcome) -> bool:
        """Verify outcome has sufficient evidence."""
        
        # Need at least one piece of evidence
        if not outcome.evidence:
            return False
        
        # Need positive verification signal
        evidence_indicators = [
            "resolved", "solved", "fixed", "working",
            "success", "complete", "passed"
        ]
        
        evidence_text = " ".join(outcome.evidence).lower()
        
        # Check for positive indicators
        has_positive = any(
            indicator in evidence_text
            for indicator in evidence_indicators
        )
        
        return has_positive
    
    def _calculate_adjustment(
        self,
        outcome: Outcome,
        similar: list[Outcome],
    ) -> float:
        """Calculate recommended confidence adjustment."""
        
        if not similar:
            # No similar cases, use this outcome alone
            if outcome.outcome_type == OutcomeType.SUCCESS:
                return 0.05  # Slight boost
            elif outcome.outcome_type == OutcomeType.FAILURE:
                return -0.10  # Significant decrease
            return 0.0
        
        # Calculate based on similar outcomes
        success_count = sum(
            1 for o in similar
            if o.outcome_type == OutcomeType.SUCCESS
        )
        
        success_ratio = success_count / len(similar)
        
        # If current outcome differs from trend
        if outcome.outcome_type == OutcomeType.SUCCESS and success_ratio < 0.5:
            # Positive outcome in negative context
            return 0.10  # Significant boost
        elif outcome.outcome_type == OutcomeType.FAILURE and success_ratio > 0.5:
            # Negative outcome in positive context
            return -0.15  # Significant decrease
        
        return 0.0  # Outcome consistent with trend
```
