class FeedbackType(Enum):
    """Types of user feedback."""
    CORRECT = "correct"                    # Recommendation was accurate
    INCORRECT = "incorrect"              # Recommendation was wrong
    PARTIALLY_CORRECT = "partially_correct"  # Some aspects correct
    MISSING_CONTEXT = "missing_context"  # More info needed
    NEEDS_REVIEW = "needs_review"        # Requires human validation
    VERY_HELPFUL = "very_helpful"        # Exceptional recommendation
    NOT_HELPFUL = "not_helpful"          # Poor recommendation


@dataclass
class Feedback:
    """User feedback on a decision."""
    feedback_id: str
    decision_id: str
    feedback_type: FeedbackType
    source: str  # "biomedical_engineer", "technician", etc.
    timestamp: datetime
    comment: str
    rating: int  # 1-5
    context: dict  # Additional context


@dataclass
class ExperienceCase:
    """Stored experience case."""
    case_id: str
    decision_id: str
    device_id: str
    device_type: str
    symptoms: list[str]
    diagnosis: str
    decision_made: str
    outcome: OutcomeType
    feedback: FeedbackType | None
    feedback_comment: str | None
    confidence_at_decision: float
    actual_confidence_after: float | None
    resolution_time: int | None
    timestamp: datetime
    engineer: str | None


class FeedbackManager:
    """Manages feedback collection and processing."""
    
    async def process(self, feedback: Feedback) -> dict:
        """Process incoming feedback."""
        
        # Validate feedback
        if not self._validate(feedback):
            return {"valid": False, "reason": "Invalid feedback"}
        
        # Extract sentiment
        sentiment = self._analyze_sentiment(feedback.comment)
        
        # Assign priority
        priority = self._assign_priority(feedback)
        
        # Generate insights
        insights = self._generate_insights(feedback)
        
        return {
            "valid": True,
            "feedback": feedback,
            "sentiment": sentiment,
            "priority": priority,
            "insights": insights,
        }
    
    def _validate(self, feedback: Feedback) -> bool:
        """Validate feedback has required fields."""
        return (
            feedback.feedback_id and
            feedback.decision_id and
            feedback.feedback_type and
            feedback.timestamp
        )
    
    def _analyze_sentiment(self, comment: str) -> str:
        """Simple sentiment analysis."""
        positive = ["great", "excellent", "correct", "perfect", "helpful"]
        negative = ["wrong", "bad", "incorrect", "poor", "failed"]
        
        comment_lower = comment.lower()
        
        positive_count = sum(1 for w in positive if w in comment_lower)
        negative_count = sum(1 for w in negative if w in comment_lower)
        
        if positive_count > negative_count:
            return "positive"
        elif negative_count > positive_count:
            return "negative"
        return "neutral"
    
    def _assign_priority(self, feedback: Feedback) -> str:
        """Assign processing priority."""
        if feedback.feedback_type == FeedbackType.INCORRECT:
            return "high"
        elif feedback.feedback_type == FeedbackType.NEEDS_REVIEW:
            return "medium"
        return "low"


class ExperienceRepository:
    """Stores and retrieves experience cases."""
    
    def __init__(self):
        self.cases: list[ExperienceCase] = []
    
    async def store(
        self,
        decision: ClinicalDecision,
        outcome: CollectedOutcome,
        feedback: Feedback | None,
    ) -> ExperienceCase:
        """Store experience case."""
        
        case = ExperienceCase(
            case_id=f"case_{datetime.now().timestamp()}",
            decision_id=decision.id,
            device_id=decision.audit.decision_id,
            device_type="unknown",
            symptoms=[],
            diagnosis=decision.decision,
            decision_made=decision.decision,
            outcome=outcome.outcome.outcome_type,
            feedback=feedback.feedback_type if feedback else None,
            feedback_comment=feedback.comment if feedback else None,
            confidence_at_decision=decision.confidence,
            actual_confidence_after=None,
            resolution_time=outcome.outcome.resolution_time_minutes,
            timestamp=datetime.now(),
            engineer=feedback.source if feedback else None,
        )
        
        self.cases.append(case)
        return case
    
    async def find_similar(
        self,
        case: ExperienceCase,
        limit: int = 5,
    ) -> list[ExperienceCase]:
        """Find similar cases."""
        
        # Simple similarity: same device type and outcome
        similar = [
            c for c in self.cases
            if c.device_type == case.device_type
            and c.outcome == case.outcome
            and c.case_id != case.case_id
        ]
        
        return similar[:limit]
