# ADR-3051: Reasoning Graph Design

**Título:** Diseño del Grafo de Razonamiento
**Estado:** ✅ COMPLETO

## Contexto

El Reasoning Graph visualiza el razonamiento de EREN como un grafo dirigido acíclico (DAG), mostrando relaciones entre conceptos, evidencia y conclusiones.

---

## Decisión

### Modelo del Grafo

```python
@dataclass(frozen=True)
class GraphNode:
    """Node in reasoning graph."""
    node_id: str
    label: str
    node_type: NodeType  # SYMPTOM, EVIDENCE, HYPOTHESIS, CONCLUSION
    properties: dict
    confidence: float | None
    source: str | None


class NodeType(Enum):
    """Types of nodes in reasoning graph."""
    SYMPTOM = "symptom"
    EVIDENCE = "evidence"
    RULE = "rule"
    HYPOTHESIS = "hypothesis"
    CONCLUSION = "conclusion"
    ACTION = "action"


@dataclass(frozen=True)
class GraphEdge:
    """Edge in reasoning graph."""
    edge_id: str
    source_id: str
    target_id: str
    edge_type: EdgeType
    weight: float
    label: str | None


class EdgeType(Enum):
    """Types of edges."""
    SUPPORTS = "supports"
    CONTRADICTS = "contradicts"
    CAUSES = "causes"
    IMPLIES = "implies"
    LEADS_TO = "leads_to"


@dataclass(frozen=True)
class ReasoningGraph:
    """Complete reasoning graph."""
    graph_id: str
    nodes: list[GraphNode]
    edges: list[GraphEdge]
    root_nodes: list[str]
    leaf_nodes: list[str]
    critical_path: list[str]


class ReasoningGraphGenerator:
    """Generates reasoning graphs."""
    
    def __init__(self):
        self.graph_analyzer = GraphAnalyzer()
    
    async def generate(
        self,
        reasoning_chain: ReasoningChain,
    ) -> ReasoningGraph:
        """Generate reasoning graph from chain."""
        
        nodes = []
        edges = []
        
        # Create nodes from chain steps
        for i, step in enumerate(reasoning_chain.steps):
            node = GraphNode(
                node_id=f"node_{i}",
                label=step.description,
                node_type=self._get_node_type(step),
                properties=step.properties,
                confidence=step.confidence,
                source=step.source,
            )
            nodes.append(node)
        
        # Create edges from step relationships
        for i in range(len(nodes) - 1):
            edge = GraphEdge(
                edge_id=f"edge_{i}",
                source_id=nodes[i].node_id,
                target_id=nodes[i + 1].node_id,
                edge_type=EdgeType.IMPLIES,
                weight=0.9,
                label=None,
            )
            edges.append(edge)
        
        # Find critical path
        critical_path = self.graph_analyzer.find_critical_path(
            nodes, edges
        )
        
        return ReasoningGraph(
            graph_id=f"graph_{datetime.now().timestamp()}",
            nodes=nodes,
            edges=edges,
            root_nodes=[nodes[0].node_id] if nodes else [],
            leaf_nodes=[nodes[-1].node_id] if nodes else [],
            critical_path=critical_path,
        )
    
    def _get_node_type(self, step: Step) -> NodeType:
        """Determine node type from step."""
        if step.type == "symptom":
            return NodeType.SYMPTOM
        elif step.type == "evidence":
            return NodeType.EVIDENCE
        elif step.type == "rule":
            return NodeType.RULE
        elif step.type == "hypothesis":
            return NodeType.HYPOTHESIS
        elif step.type == "conclusion":
            return NodeType.CONCLUSION
        elif step.type == "action":
            return NodeType.ACTION
        return NodeType.HYPOTHESIS
```

### Exportación DOT

```python
class GraphVisualizer:
    """Visualizes reasoning graphs."""
    
    def to_dot(self, graph: ReasoningGraph) -> str:
        """Export to DOT format."""
        lines = [
            "digraph ReasoningGraph {",
            "  rankdir=TB;",
            "  node [shape=box];",
        ]
        
        # Add nodes
        for node in graph.nodes:
            color = self._get_node_color(node.node_type)
            lines.append(
                f'  {node.node_id} [label="{node.label}" '
                f'fillcolor={color} style=filled];'
            )
        
        # Add edges
        for edge in graph.edges:
            style = "dashed" if edge.edge_type == EdgeType.CONTRADICTS else "solid"
            lines.append(
                f"  {edge.source_id} -> {edge.target_id} "
                f"[style={style} label={edge.weight:.2f}];"
            )
        
        lines.append("}")
        return "\n".join(lines)
    
    def _get_node_color(self, node_type: NodeType) -> str:
        """Get color for node type."""
        colors = {
            NodeType.SYMPTOM: "#FFE4B5",    # Moccasin
            NodeType.EVIDENCE: "#87CEEB",   # Sky blue
            NodeType.RULE: "#98FB98",       # Pale green
            NodeType.HYPOTHESIS: "#DDA0DD", # Plum
            NodeType.CONCLUSION: "#F0E68C", # Khaki
            NodeType.ACTION: "#FFB6C1",     # Light pink
        }
        return colors.get(node_type, "#FFFFFF")


class GraphAnalyzer:
    """Analyzes reasoning graphs."""
    
    def find_critical_path(
        self,
        nodes: list[GraphNode],
        edges: list[GraphEdge],
    ) -> list[str]:
        """Find critical path through graph."""
        # Build adjacency list
        adj = {n.node_id: [] for n in nodes}
        for edge in edges:
            adj[edge.source_id].append(edge.target_id)
        
        # Simple path from root to leaf
        path = []
        if nodes:
            current = nodes[0].node_id
            path.append(current)
            while adj[current]:
                current = adj[current][0]
                path.append(current)
        
        return path
```
