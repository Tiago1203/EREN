# ADR-3052: Evidence Tree Design

**Título:** Diseño del Árbol de Evidencia
**Estado:** ✅ COMPLETO
**Fecha:** 2026-07-21
**Decisor:** Architecture Team

---

## Contexto

El Evidence Tree organiza jerárquicamente toda la evidencia relacionada con una recomendación, mostrando qué apoya y qué contradice.

---

## Decisión

### Modelo del Árbol

```python
@dataclass(frozen=True)
class TreeNode:
    """Node in evidence tree."""
    node_id: str
    label: str
    node_type: TreeNodeType  # ROOT, CATEGORY, EVIDENCE
    children: list[TreeNode]
    annotation: NodeAnnotation | None
    confidence: float
    source: str | None


class TreeNodeType(Enum):
    """Types of tree nodes."""
    ROOT = "root"
    CATEGORY = "category"
    EVIDENCE = "evidence"
    CONCLUSION = "conclusion"


@dataclass(frozen=True)
class NodeAnnotation:
    """Annotation for tree node."""
    weight: float
    source_type: str
    citation: str | None
    relevance: float


@dataclass(frozen=True)
class EvidenceTree:
    """Complete evidence tree."""
    tree_id: str
    root: TreeNode
    supporting_evidence: list[TreeNode]
    contradicting_evidence: list[TreeNode]
    total_evidence_count: int
    overall_weight: float


class EvidenceTreeBuilder:
    """Builds evidence trees."""
    
    async def build(
        self,
        evidence_bundle: EvidenceBundle,
    ) -> EvidenceTree:
        """Build evidence tree from bundle."""
        
        # Create root node
        root = TreeNode(
            node_id="root",
            label="Recommendation",
            node_type=TreeNodeType.ROOT,
            children=[],
            annotation=None,
            confidence=0.9,
            source=None,
        )
        
        # Create category nodes
        categories = self._categorize_evidence(evidence_bundle)
        
        supporting_nodes = []
        contradicting_nodes = []
        
        for category_name, evidence_list in categories.items():
            category_node = self._build_category_node(
                category_name, evidence_list
            )
            root.children.append(category_node)
            
            for e in evidence_list:
                if e.get("supports", True):
                    supporting_nodes.append(e)
                else:
                    contradicting_nodes.append(e)
        
        # Calculate overall weight
        overall_weight = self._calculate_overall_weight(evidence_bundle)
        
        return EvidenceTree(
            tree_id=f"tree_{datetime.now().timestamp()}",
            root=root,
            supporting_evidence=supporting_nodes,
            contradicting_evidence=contradicting_nodes,
            total_evidence_count=len(supporting_nodes) + len(contradicting_nodes),
            overall_weight=overall_weight,
        )
    
    def _categorize_evidence(
        self,
        bundle: EvidenceBundle,
    ) -> dict[str, list]:
        """Categorize evidence by type."""
        categories = {
            "Standards & Regulations": [],
            "Manufacturer Guidelines": [],
            "Historical Incidents": [],
            "Technical Analysis": [],
            "Clinical Evidence": [],
        }
        
        for evidence in bundle.supporting_evidence:
            category = self._classify_evidence(evidence)
            categories[category].append(evidence)
        
        for evidence in bundle.contradicting_evidence:
            category = self._classify_evidence(evidence)
            categories[category].append(evidence)
        
        return categories
    
    def _classify_evidence(self, evidence: EvidenceItem) -> str:
        """Classify evidence into category."""
        source = evidence.source_type.value
        
        if source in ["standard", "regulation"]:
            return "Standards & Regulations"
        elif source in ["manual", "manufacturer"]:
            return "Manufacturer Guidelines"
        elif source == "historical":
            return "Historical Incidents"
        elif source == "knowledge_graph":
            return "Technical Analysis"
        else:
            return "Clinical Evidence"
    
    def _build_category_node(
        self,
        category_name: str,
        evidence_list: list,
    ) -> TreeNode:
        """Build category node with evidence children."""
        children = []
        
        for i, evidence in enumerate(evidence_list):
            child = TreeNode(
                node_id=f"ev_{category_name}_{i}",
                label=evidence.content[:100],  # Truncate
                node_type=TreeNodeType.EVIDENCE,
                children=[],
                annotation=NodeAnnotation(
                    weight=evidence.quality_score,
                    source_type=evidence.source_type.value,
                    citation=evidence.citation,
                    relevance=evidence.relevance_score,
                ),
                confidence=evidence.quality_score * evidence.relevance_score,
                source=evidence.source_name,
            )
            children.append(child)
        
        return TreeNode(
            node_id=f"cat_{category_name}",
            label=category_name,
            node_type=TreeNodeType.CATEGORY,
            children=children,
            annotation=None,
            confidence=sum(e.quality_score for e in evidence_list) / len(evidence_list) if evidence_list else 0,
            source=None,
        )
    
    def _calculate_overall_weight(self, bundle: EvidenceBundle) -> float:
        """Calculate overall evidence weight."""
        supporting = bundle.supporting_evidence
        contradicting = bundle.contradicting_evidence
        
        if not supporting:
            return 0.2
        
        supporting_weight = sum(
            e.quality_score * e.relevance_score 
            for e in supporting
        ) / len(supporting)
        
        contradiction_penalty = len(contradicting) * 0.1
        
        return max(0, supporting_weight - contradiction_penalty)
```

### Exportación

```python
class TreeExporter:
    """Exports evidence trees."""
    
    def to_json(self, tree: EvidenceTree) -> str:
        """Export to JSON."""
        import json
        return json.dumps(self._tree_to_dict(tree), indent=2)
    
    def to_markdown(self, tree: EvidenceTree) -> str:
        """Export to Markdown."""
        lines = ["# Evidence Tree", ""]
        lines.extend(self._node_to_markdown(tree.root, 0))
        return "\n".join(lines)
    
    def _node_to_markdown(self, node: TreeNode, level: int) -> list[str]:
        """Convert node to markdown."""
        lines = []
        indent = "  " * level
        bullet = "•" if node.node_type == TreeNodeType.EVIDENCE else "#"
        
        annotation_str = ""
        if node.annotation:
            annotation_str = f" [weight={node.annotation.weight:.2f}]"
        
        lines.append(f"{indent}{bullet} {node.label}{annotation_str}")
        
        for child in node.children:
            lines.extend(self._node_to_markdown(child, level + 1))
        
        return lines
    
    def _tree_to_dict(self, tree: EvidenceTree) -> dict:
        """Convert tree to dictionary."""
        return {
            "tree_id": tree.tree_id,
            "root": self._node_to_dict(tree.root),
            "total_evidence": tree.total_evidence_count,
            "overall_weight": tree.overall_weight,
        }
    
    def _node_to_dict(self, node: TreeNode) -> dict:
        """Convert node to dictionary."""
        return {
            "node_id": node.node_id,
            "label": node.label,
            "type": node.node_type.value,
            "confidence": node.confidence,
            "children": [self._node_to_dict(c) for c in node.children],
        }
```
