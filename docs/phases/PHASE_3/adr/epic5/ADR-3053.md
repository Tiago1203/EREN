```python
class LanguageStyle(Enum):
    """Styles for natural language generation."""
    TECHNICAL = "technical"     # Highly technical
    CLINICAL = "clinical"      # Clinical terminology
    SIMPLE = "simple"          # Layperson friendly
    FORMAL = "formal"          # Formal documentation


class NaturalLanguageExplainer:
    """Generates natural language explanations."""
    
    def __init__(self):
        self.templates = self._load_templates()
    
    async def generate(
        self,
        recommendation: Recommendation,
        evidence_bundle: EvidenceBundle,
        confidence_score: ConfidenceScore,
        style: LanguageStyle = LanguageStyle.CLINICAL,
    ) -> str:
        """Generate natural language explanation."""
        
        # Build sections
        sections = []
        
        # Header
        sections.append(self._generate_header(recommendation))
        
        # Why section
        sections.append(self._generate_why_section(evidence_bundle, style))
        
        # Evidence section
        sections.append(self._generate_evidence_section(evidence_bundle, style))
        
        # Confidence section
        sections.append(self._generate_confidence_section(confidence_score))
        
        # Sources section
        sections.append(self._generate_sources_section(evidence_bundle))
        
        return "\n\n".join(sections)
    
    def _generate_header(
        self,
        recommendation: Recommendation,
    ) -> str:
        """Generate header section."""
        return f"""EREN recommends: {recommendation.action.upper()}"""
    
    def _generate_why_section(
        self,
        bundle: EvidenceBundle,
        style: LanguageStyle,
    ) -> str:
        """Generate 'Why' section."""
        lines = ["Why?", "━" * 50, ""]
        
        supporting = bundle.supporting_evidence[:3]  # Top 3
        
        for i, evidence in enumerate(supporting, 1):
            if style == LanguageStyle.SIMPLE:
                explanation = self._simplify_evidence(evidence)
            elif style == LanguageStyle.TECHNICAL:
                explanation = self._technical_evidence(evidence)
            else:
                explanation = self._clinical_evidence(evidence)
            
            lines.append(f"{i}. {evidence.content}")
            lines.append(f"   Source: {evidence.source_name}")
            if evidence.citation:
                lines.append(f"   Reference: {evidence.citation}")
            lines.append("")
        
        return "\n".join(lines)
    
    def _generate_evidence_section(
        self,
        bundle: EvidenceBundle,
        style: LanguageStyle,
    ) -> str:
        """Generate detailed evidence section."""
        lines = ["Evidence Details:", "━" * 50, ""]
        
        # Group by source type
        by_source = {}
        for e in bundle.supporting_evidence:
            source = e.source_type.value
            if source not in by_source:
                by_source[source] = []
            by_source[source].append(e)
        
        for source_type, evidence_list in by_source.items():
            lines.append(f"\n{source_type.upper().replace('_', ' ')}:")
            for e in evidence_list:
                lines.append(f"  • {e.content[:80]}...")
        
        return "\n".join(lines)
    
    def _generate_confidence_section(
        self,
        confidence: ConfidenceScore,
    ) -> str:
        """Generate confidence section."""
        level_emoji = {
            "high": "✓✓",
            "medium": "✓",
            "low": "⚠",
            "uncertain": "⚠⚠",
        }
        
        emoji = level_emoji.get(confidence.level.value, "•")
        
        lines = [
            "",
            f"Confidence: {confidence.percentage}% ({confidence.level.value.upper()}) {emoji}",
            "",
        ]
        
        if confidence.supporting_factors:
            lines.append("Supporting factors:")
            for factor in confidence.supporting_factors:
                lines.append(f"  ✓ {factor}")
        
        if confidence.uncertainty_factors:
            lines.append("")
            lines.append("Uncertainty factors:")
            for factor in confidence.uncertainty_factors:
                lines.append(f"  ⚠ {factor}")
        
        return "\n".join(lines)
    
    def _generate_sources_section(
        self,
        bundle: EvidenceBundle,
    ) -> str:
        """Generate sources section."""
        lines = ["", "Sources:", "━" * 50, ""]
        
        citations = []
        for e in bundle.supporting_evidence:
            if e.citation:
                citations.append(e.citation)
            else:
                citations.append(f"{e.source_name}")
        
        for i, citation in enumerate(set(citations), 1):
            lines.append(f"{i}. {citation}")
        
        return "\n".join(lines)
    
    def _simplify_evidence(self, evidence) -> str:
        """Simplify evidence for layperson."""
        # Replace technical terms with simple ones
        simplifications = {
            "leakage current": "electrical safety issue",
            "calibration": "proper setting check",
            "hypothesis": "possible cause",
            "algorithm": "computer calculation",
        }
        
        content = evidence.content.lower()
        for technical, simple in simplifications.items():
            content = content.replace(technical, simple)
        
        return content.capitalize()
    
    def _technical_evidence(self, evidence) -> str:
        """Make evidence more technical."""
        return f"[TECHNICAL] {evidence.content}"
    
    def _clinical_evidence(self, evidence) -> str:
        """Make evidence clinically appropriate."""
        return evidence.content
```

### Plantillas

```python
TEMPLATES = {
    "header": "EREN recommends: {action}",
    "why_header": "Why?",
    "evidence_header": "Evidence:",
    "confidence_header": "Confidence Level:",
    "sources_header": "References:",
    "uncertain_recommendation": "More investigation may be needed.",
    "certain_recommendation": "This recommendation is well-supported.",
}


class ExplanationTemplate:
    """Template-based explanation generator."""
    
    def apply(
        self,
        template_key: str,
        context: dict,
    ) -> str:
        """Apply template with context."""
        template = TEMPLATES.get(template_key, "")
        return template.format(**context)
```
