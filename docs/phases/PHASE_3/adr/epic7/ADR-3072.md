# ADR-3072: Hazard Detection Design

**Título:** Diseño de Detección de Peligros
**Estado:** ✅ COMPLETO

**Decisor:** Architecture Team

## Contexto

Detección proactiva de peligros en equipos biomédicos.

---

## Decisión

### Categorías de Peligros

```
Hazard Categories
├── Electrical
│   ├── Shock hazard (leakage > 0.5mA)
│   ├── Fire hazard (overheating)
│   └── Explosion hazard (spark in O2 env)
├── Radiation
│   ├── X-ray exposure
│   ├── UV exposure
│   └── Laser hazard
├── Biological
│   ├── Infection risk
│   ├── Cross-contamination
│   └── Biohazard exposure
├── Mechanical
│   ├── Crush hazard
│   ├── Pinch hazard
│   └── Fall hazard
└── Chemical
    ├── Toxic exposure
    ├── Corrosion
    └── Fire risk
```

### Detector de Peligros

```python
class HazardType(Enum):
    """Types of hazards."""
    ELECTRICAL_SHOCK = "electrical_shock"
    ELECTRICAL_FIRE = "electrical_fire"
    RADIATION_EXPOSURE = "radiation_exposure"
    INFECTION_RISK = "infection_risk"
    MECHANICAL_INJURY = "mechanical_injury"
    CHEMICAL_EXPOSURE = "chemical_exposure"


@dataclass
class Hazard:
    """Detected hazard."""
    hazard_id: str
    hazard_type: HazardType
    severity: Severity
    description: str
    evidence: list[str]
    mitigation: str


class HazardDetector:
    """Detects hazards in equipment/context."""
    
    def detect(self, context: dict) -> list[SafetyBlocker]:
        """Detect all hazards."""
        blockers = []
        
        # Electrical hazards
        electrical = self._detect_electrical_hazards(context)
        blockers.extend(electrical)
        
        # Radiation hazards
        radiation = self._detect_radiation_hazards(context)
        blockers.extend(radiation)
        
        # Biological hazards
        biological = self._detect_biological_hazards(context)
        blockers.extend(biological)
        
        return blockers
    
    def _detect_electrical_hazards(
        self,
        context: dict,
    ) -> list[SafetyBlocker]:
        """Detect electrical hazards."""
        blockers = []
        
        # Check leakage current
        leakage = context.get("leakage_current_ma", 0)
        if leakage > 0.5:
            blockers.append(SafetyBlocker(
                blocker_id="hazard_electrical_shock",
                category=SafetyCategory.ELECTRICAL_SAFETY,
                reason=f"Leakage current {leakage}mA exceeds safe limit of 0.5mA",
                severity=Severity.CRITICAL,
                action="IMMEDIATE REMOVAL FROM SERVICE",
                alternative=None,
            ))
        
        # Check for damaged insulation
        if context.get("insulation_damaged", False):
            blockers.append(SafetyBlocker(
                blocker_id="hazard_insulation",
                category=SafetyCategory.ELECTRICAL_SAFETY,
                reason="Damaged insulation creates shock hazard",
                severity=Severity.CRITICAL,
                action="IMMEDIATE REMOVAL FROM SERVICE",
                alternative=None,
            ))
        
        # Check for missing ground
        if context.get("ground_missing", False):
            blockers.append(SafetyBlocker(
                blocker_id="hazard_ground",
                category=SafetyCategory.ELECTRICAL_SAFETY,
                reason="Missing protective ground creates shock hazard",
                severity=Severity.CRITICAL,
                action="IMMEDIATE REMOVAL FROM SERVICE",
                alternative=None,
            ))
        
        return blockers
    
    def _detect_radiation_hazards(
        self,
        context: dict,
    ) -> list[SafetyBlocker]:
        """Detect radiation hazards."""
        blockers = []
        
        # Check X-ray equipment
        if context.get("equipment_type") == "xray":
            if not context.get("radiation_shielding_intact", True):
                blockers.append(SafetyBlocker(
                    blocker_id="hazard_radiation_shielding",
                    category=SafetyCategory.RADIATION_SAFETY,
                    reason="Radiation shielding is damaged",
                    severity=Severity.CRITICAL,
                    action="Remove from service, radiation safety review required",
                    alternative=None,
                ))
            
            if not context.get("dose_monitoring_active", True):
                blockers.append(SafetyBlocker(
                    blocker_id="hazard_radiation_dosimetry",
                    category=SafetyCategory.RADIATION_SAFETY,
                    reason="Dose monitoring not active",
                    severity=Severity.HIGH,
                    action="Activate dose monitoring",
                    alternative=None,
                ))
        
        return blockers
    
    def _detect_biological_hazards(
        self,
        context: dict,
    ) -> list[SafetyBlocker]:
        """Detect biological hazards."""
        blockers = []
        
        # Check for contaminated equipment
        if context.get("equipment_contaminated", False):
            biohazard_type = context.get("biohazard_type", "unknown")
            blockers.append(SafetyBlocker(
                blocker_id="hazard_biohazard",
                category=SafetyCategory.BIOLOGICAL_SAFETY,
                reason=f"Equipment contaminated with {biohazard_type}",
                severity=Severity.CRITICAL,
                action="Decontaminate before use or dispose per protocol",
                alternative=None,
            ))
        
        # Check for single-use device reuse
        if context.get("single_use_reused", False):
            blockers.append(SafetyBlocker(
                blocker_id="hazard_single_use",
                category=SafetyCategory.BIOLOGICAL_SAFETY,
                reason="Single-use device has been reused",
                severity=Severity.CRITICAL,
                action="Do not use - dispose per single-use policy",
                alternative="Use new single-use device",
            ))
        
        return blockers
```
