## Decisión

### Acciones Bloqueadas (Hard Blocks)

```python
class BlockedAction:
    """Blocked action definition."""
    action_pattern: str
    category: SafetyCategory
    severity: Severity
    reason: str
    alternative: str | None


BLOCKED_ACTIONS = [
    # Equipment操作
    BlockedAction(
        action_pattern=r"open.*defibrillator",
        category=SafetyCategory.EQUIPMENT,
        severity=Severity.CRITICAL,
        reason="Opening defibrillator while charged can cause lethal shock",
        alternative="Use manufacturer discharge procedure",
    ),
    BlockedAction(
        action_pattern=r"disable.*alarm",
        category=SafetyCategory.PATIENT_SAFETY,
        severity=Severity.CRITICAL,
        reason="Disabling alarms can cause missed critical events",
        alternative="Configure alarm limits appropriately",
    ),
    BlockedAction(
        action_pattern=r"bypass.*isolation",
        category=SafetyCategory.ELECTRICAL_SAFETY,
        severity=Severity.CRITICAL,
        reason="Bypassing isolation can cause shock hazard",
        alternative=None,
    ),
    BlockedAction(
        action_pattern=r"override.*safety",
        category=SafetyCategory.SAFETY_SYSTEM,
        severity=Severity.CRITICAL,
        reason="Safety system override can cause equipment damage",
        alternative=None,
    ),
    BlockedAction(
        action_pattern=r"disable.*ground",
        category=SafetyCategory.ELECTRICAL_SAFETY,
        severity=Severity.CRITICAL,
        reason="Disabling ground can cause shock hazard",
        alternative=None,
    ),
    # Authentication bypasses
    BlockedAction(
        action_pattern=r"bypass.*authentication",
        category=SafetyCategory.SECURITY,
        severity=Severity.HIGH,
        reason="Authentication bypass violates security policy",
        alternative=None,
    ),
    BlockedAction(
        action_pattern=r"disable.*audit",
        category=SafetyCategory.COMPLIANCE,
        severity=Severity.HIGH,
        reason="Disabling audit violates compliance requirements",
        alternative=None,
    ),
]


class SafetyValidator:
    """Validates safety of recommendations."""
    
    def check_blocked_actions(
        self,
        recommendation: Recommendation,
    ) -> list[SafetyBlocker]:
        """Check if action is blocked."""
        blockers = []
        action_text = recommendation.action.lower()
        
        for blocked in BLOCKED_ACTIONS:
            if re.search(blocked.action_pattern, action_text, re.IGNORECASE):
                blockers.append(SafetyBlocker(
                    blocker_id=f"blocked_{blocked.category.value}",
                    category=blocked.category,
                    reason=blocked.reason,
                    severity=blocked.severity,
                    action=f"Avoid: {action_text}",
                    alternative=blocked.alternative,
                ))
        
        return blockers
    
    def check_clinical_safety(
        self,
        context: dict,
    ) -> list[SafetyWarning]:
        """Check clinical safety constraints."""
        warnings = []
        
        # Check alarm configuration
        if context.get("alarms_disabled"):
            warnings.append(SafetyWarning(
                warning_id="clinical_alarm_disabled",
                category=SafetyCategory.PATIENT_SAFETY,
                description="Alarms are currently disabled",
                recommendation="Enable alarms or configure appropriate limits",
            ))
        
        # Check patient monitoring
        if context.get("patient_unmonitored"):
            warnings.append(SafetyWarning(
                warning_id="clinical_patient_unmonitored",
                category=SafetyCategory.PATIENT_SAFETY,
                description="Patient is currently unmonitored",
                recommendation="Ensure adequate monitoring is in place",
            ))
        
        return warnings
```
