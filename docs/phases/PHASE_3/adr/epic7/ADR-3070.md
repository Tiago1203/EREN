### Modelo de Seguridad

```python
class SafetyDecision(Enum):
    """Safety decision outcomes."""
    ALLOW = "allow"
    ALLOW_WITH_WARNING = "allow_with_warning"
    BLOCK = "block"
    ESCALATE = "escalate"


@dataclass
class SafetyResult:
    """Result of safety validation."""
    decision: SafetyDecision
    is_safe: bool
    blockers: list[SafetyBlocker]
    warnings: list[SafetyWarning]
    risk_score: float
    risk_level: RiskLevel
    recommendations: list[str]
    requires_human_review: bool
    critical_alert_triggered: bool


@dataclass
class SafetyBlocker:
    """Safety blocker for dangerous actions."""
    blocker_id: str
    category: SafetyCategory
    reason: str
    severity: Severity
    action: str
    alternative: str | None


class SafetyEngine:
    """Main safety engine."""
    
    def __init__(self):
        self.validator = SafetyValidator()
        self.risk_analyzer = RiskAnalyzer()
        self.hazard_detector = HazardDetector()
        self.alert_system = CriticalAlertSystem()
        self.override_system = HumanOverrideSystem()
    
    async def validate(
        self,
        recommendation: Recommendation,
        context: dict,
    ) -> SafetyResult:
        """Validate recommendation for safety."""
        
        blockers = []
        warnings = []
        risk_score = 0.0
        
        # Check blocked actions
        blocked = self.validator.check_blocked_actions(recommendation)
        if blocked:
            blockers.extend(blocked)
            risk_score = max(risk_score, 0.9)
        
        # Check clinical safety
        clinical_issues = self.validator.check_clinical_safety(context)
        if clinical_issues:
            warnings.extend(clinical_issues)
            risk_score = max(risk_score, 0.6)
        
        # Detect hazards
        hazards = self.hazard_detector.detect(context)
        if hazards:
            blockers.extend(hazards)
            risk_score = max(risk_score, 0.95)
        
        # Determine decision
        decision = self._determine_decision(blockers, warnings, risk_score)
        
        # Handle critical alerts
        critical = self._should_trigger_alert(blockers, risk_score)
        
        return SafetyResult(
            decision=decision,
            is_safe=decision == SafetyDecision.ALLOW,
            blockers=blockers,
            warnings=warnings,
            risk_score=risk_score,
            risk_level=self._classify_risk(risk_score),
            recommendations=self._get_recommendations(decision),
            requires_human_review=decision == SafetyDecision.ESCALATE,
            critical_alert_triggered=critical,
        )
```
