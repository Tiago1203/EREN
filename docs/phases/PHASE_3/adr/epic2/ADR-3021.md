# ADR-3021: Hypothesis Engine Design

**Título:** Diseño del Motor de Hipótesis
**Estado:** ✅ COMPLETO
**Fecha:** 2026-07-21
**Decisor:** Architecture Team

---

## Contexto

El Hypothesis Engine genera, evalúa y prioriza hipótesis a partir de síntomas y evidencia. Es el primer paso del pipeline de razonamiento.

Recibe: síntomas, contexto, evidencia
Produce: hipótesis ranked con probabilidades

---

## Decisión

### Arquitectura del Hypothesis Engine

```
HypothesisEngine
├── HypothesisGenerator
│   ├── EquipmentBasedGenerator
│   │   # Genera hipótesis basadas en tipo de equipo
│   ├── PatternBasedGenerator
│   │   # Genera hipótesis basadas en patrones conocidos
│   └── EvidenceBasedGenerator
│       # Genera hipótesis basadas en evidencia disponible
│
├── HypothesisEvaluator
│   ├── ProbabilityCalculator
│   │   # Calcula P(H|E) usando Bayes
│   ├── EvidenceAssessor
│   │   # Evalúa peso de evidencia
│   └── ConsistencyChecker
│       # Verifica consistencia entre hipótesis
│
└── HypothesisPrioritizer
    ├── SeverityRanker
    │   # Ordena por severidad
    ├── LikelihoodRanker
    │   # Ordena por probabilidad
    └── CompositeRanker
        # Combina múltiples criterios
```

### Modelo de Datos

```python
@dataclass(frozen=True)
class Hypothesis:
    """Hipótesis generada."""
    hypothesis_id: str
    name: str
    description: str
    probability: float
    confidence: float
    supporting_evidence: list[Evidence]
    contradicting_evidence: list[Evidence]
    related_symptoms: list[str]
    equipment_category: str | None
    severity: SeverityLevel
    created_at: datetime
    metadata: dict

@dataclass(frozen=True)
class HypothesisSet:
    """Conjunto de hipótesis relacionadas."""
    set_id: str
    hypotheses: list[Hypothesis]
    top_hypothesis: Hypothesis
    differential_diagnosis: list[Hypothesis]
    mutual_exclusivity: bool
    collectively_exhaustive: bool

class HypothesisEngine:
    """Motor de generación y evaluación de hipótesis."""
    
    async def generate_hypotheses(
        self,
        context: ReasoningContext,
        symptoms: list[Symptom],
    ) -> HypothesisSet:
        """Genera conjunto de hipótesis a partir de síntomas."""
        # 1. Generar candidatos
        candidates = await self.generator.generate(symptoms, context)
        
        # 2. Recolectar evidencia
        evidence = await self.collect_evidence(candidates, context)
        
        # 3. Evaluar probabilidades
        evaluated = await self.evaluator.evaluate(candidates, evidence)
        
        # 4. Priorizar
        ranked = await self.prioritizer.rank(evaluated)
        
        return HypothesisSet(
            set_id=f"hs_{datetime.now().timestamp()}",
            hypotheses=ranked,
            top_hypothesis=ranked[0] if ranked else None,
            differential_diagnosis=ranked[:5],
            mutual_exclusivity=True,
            collectively_exhaustive=True,
        )
    
    async def update_probability(
        self,
        hypothesis: Hypothesis,
        new_evidence: Evidence,
    ) -> Hypothesis:
        """Actualiza probabilidad de hipótesis con nueva evidencia."""
        # P(H|E) = P(E|H) * P(H) / P(E)
        prior = hypothesis.probability
        
        likelihood = await self.calculate_likelihood(hypothesis, new_evidence)
        evidence_prior = await self.calculate_evidence_prior(new_evidence)
        
        posterior = (likelihood * prior) / evidence_prior if evidence_prior > 0 else prior
        
        return Hypothesis(
            hypothesis_id=hypothesis.hypothesis_id,
            name=hypothesis.name,
            description=hypothesis.description,
            probability=posterior,
            confidence=hypothesis.confidence,
            supporting_evidence=hypothesis.supporting_evidence + [new_evidence],
            contradicting_evidence=hypothesis.contradicting_evidence,
            related_symptoms=hypothesis.related_symptoms,
            equipment_category=hypothesis.equipment_category,
            severity=hypothesis.severity,
            created_at=hypothesis.created_at,
            metadata=hypothesis.metadata,
        )
```

### Algoritmo de Generación

```python
async def generate_from_equipment(
    self,
    equipment_type: str,
    symptoms: list[Symptom],
    knowledge: KnowledgeContext,
) -> list[Hypothesis]:
    """Genera hipótesis basadas en tipo de equipo."""
    # 1. Obtener modos de falla del conocimiento
    failure_modes = await knowledge.get_failure_modes(equipment_type)
    
    # 2. Match symptoms con failure modes
    matched_modes = []
    for mode in failure_modes:
        if self.symptom_matches_failure_mode(symptoms, mode):
            matched_modes.append(mode)
    
    # 3. Generar hipótesis
    hypotheses = []
    for mode in matched_modes:
        hypothesis = Hypothesis(
            hypothesis_id=f"h_{mode.failure_mode_id}",
            name=mode.name,
            description=mode.description,
            probability=mode.base_probability,
            confidence=0.7,
            supporting_evidence=[],
            contradicting_evidence=[],
            related_symptoms=[s.symptom_id for s in symptoms],
            equipment_category=equipment_type,
            severity=mode.severity,
            created_at=datetime.now(),
            metadata={"source": "equipment_model"},
        )
        hypotheses.append(hypothesis)
    
    return hypotheses

def symptom_matches_failure_mode(
    self,
    symptoms: list[Symptom],
    failure_mode: FailureMode,
) -> bool:
    """Verifica si síntomas coinciden con modo de falla."""
    symptom_ids = {s.symptom_id for s in symptoms}
    mode_symptoms = set(failure_mode.symptoms)
    
    overlap = symptom_ids & mode_symptoms
    return len(overlap) / len(mode_symptoms) >= 0.5
```
