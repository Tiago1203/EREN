# ADR-3022: Inference Engine Design

**Título:** Diseño del Motor de Inferencia
**Estado:** ✅ COMPLETO
**Fecha:** 2026-07-21
**Decisor:** Architecture Team

---

## Contexto

El Inference Engine implementa múltiples tipos de razonamiento:
- Forward Chaining (encadenamiento hacia adelante)
- Backward Chaining (encadenamiento hacia atrás)
- Abductive Reasoning (razonamiento abductivo)
- Bayesian Inference (inferencia bayesiana)

---

## Decisión

### Arquitectura del Inference Engine

```
InferenceEngine
├── ForwardChaining
│   ├── RuleApplicator
│   │   # Aplica reglas a hechos conocidos
│   ├── ChainBuilder
│   │   # Construye cadenas de inferencia
│   └── ConflictResolver
│       # Resuelve conflictos entre conclusiones
│
├── BackwardChaining
│   ├── GoalDecomposer
│   │   # Descompone objetivo en sub-objetivos
│   ├── RuleSearcher
│   │   # Busca reglas que prueben objetivo
│   └── HypothesisTester
│       # Prueba hipótesis recursivamente
│
├── AbductiveReasoning
│   ├── ExplanationGenerator
│   │   # Genera explicaciones para observaciones
│   ├── BestExplanationFinder
│   │   # Encuentra mejor explicación
│   └── AbductionValidator
│       # Valida explicaciones generadas
│
└── BayesianInference
    ├── ProbabilityNetwork
    │   # Red de probabilidades
    ├── ConditionalCalculator
    │   # Calcula probabilidades condicionales
    └── PosteriorUpdater
        # Actualiza probabilidades posteriores
```

### Modelo de Datos

```python
@dataclass(frozen=True)
class InferenceRule:
    """Regla de inferencia."""
    rule_id: str
    antecedents: list[Fact]        # Condiciones necesarias
    consequents: list[Fact]         # Conclusiones
    confidence: float               # Confianza en la regla
    source: str                     # Fuente de la regla
    conditions: list[Condition]    # Condiciones adicionales

@dataclass(frozen=True)
class InferenceChain:
    """Cadena de inferencia."""
    chain_id: str
    steps: list[InferenceStep]
    conclusion: Fact
    confidence: float
    used_rules: list[str]

@dataclass(frozen=True)
class InferenceStep:
    """Paso en una cadena de inferencia."""
    step_number: int
    rule_applied: InferenceRule
    facts_used: list[Fact]
    facts_derived: list[Fact]
    confidence: float

class InferenceEngine:
    """Motor de inferencia multi-estrategia."""
    
    async def forward_chain(
        self,
        facts: list[Fact],
        rules: list[InferenceRule],
        max_iterations: int = 100,
    ) -> InferenceResult:
        """Encadenamiento hacia adelante."""
        known_facts = set(f.id for f in facts)
        inferred_facts: list[Fact] = []
        applied_rules: set[str] = set()
        
        for iteration in range(max_iterations):
            new_facts = []
            
            for rule in rules:
                if rule.rule_id in applied_rules:
                    continue
                
                if self.can_apply_rule(rule, known_facts):
                    conclusions = self.apply_rule(rule, known_facts)
                    for conclusion in conclusions:
                        if conclusion.id not in known_facts:
                            new_facts.append(conclusion)
                            known_facts.add(conclusion.id)
                    applied_rules.add(rule.rule_id)
            
            if not new_facts:
                break
            
            inferred_facts.extend(new_facts)
        
        return InferenceResult(
            inferred_facts=inferred_facts,
            applied_rules=list(applied_rules),
            iterations=iteration + 1,
        )
    
    async def backward_chain(
        self,
        goal: Fact,
        facts: list[Fact],
        rules: list[InferenceRule],
    ) -> InferenceResult:
        """Encadenamiento hacia atrás."""
        known_facts = {f.id: f for f in facts}
        sub_goals = [goal]
        proof_tree = ProofTree(goal=goal)
        
        while sub_goals:
            current_goal = sub_goals.pop(0)
            
            if current_goal.id in known_facts:
                proof_tree.add_proven(current_goal)
                continue
            
            supporting_rules = self.find_supporting_rules(current_goal, rules)
            
            for rule in supporting_rules:
                new_goals = []
                for antecedent in rule.antecedents:
                    if antecedent.id not in known_facts:
                        new_goals.append(antecedent)
                        sub_goals.append(antecedent)
                
                proof_tree.add_rule_application(rule, current_goal)
        
        return InferenceResult(
            goal_proven=goal.id in known_facts,
            proof_tree=proof_tree,
        )
    
    async def abduce(
        self,
        observation: Fact,
        knowledge: KnowledgeContext,
    ) -> list[Explanation]:
        """Razonamiento abductivo."""
        # 1. Encontrar reglas que produzcan la observación
        producing_rules = [
            r for r in knowledge.get_rules()
            if observation in r.consequents
        ]
        
        # 2. Generar explicaciones
        explanations = []
        for rule in producing_rules:
            explanation = Explanation(
                rule=rule,
                antecedent=rule.antecedents[0] if rule.antecedents else None,
                observation=observation,
                plausibility=self.calculate_plausibility(rule, knowledge),
            )
            explanations.append(explanation)
        
        # 3. Ordenar por plausibilidad
        explanations.sort(key=lambda e: e.plausibility, reverse=True)
        
        return explanations[:5]
    
    async def bayesian_update(
        self,
        hypothesis: Hypothesis,
        evidence: Evidence,
        prior_probability: float,
    ) -> float:
        """Actualización bayesiana de probabilidad."""
        likelihood = await self.calculate_likelihood(hypothesis, evidence)
        evidence_prob = await self.calculate_evidence_probability(evidence)
        
        posterior = (likelihood * prior_probability) / evidence_prob
        return min(max(posterior, 0.0), 1.0)
```

### Encadenamiento hacia Adelante

```python
async def forward_chain_example(
    self,
    symptoms: list[Symptom],
) -> list[Diagnosis]:
    """Ejemplo de encadenamiento hacia adelante."""
    facts = [
        Fact(id="symptom_1", content="low_spo2"),
        Fact(id="symptom_2", content="irregular_pulse"),
        Fact(id="equipment", content="pulse_oximeter"),
    ]
    
    rules = [
        InferenceRule(
            rule_id="rule_1",
            antecedents=[Fact(id="low_spo2", content="low_spo2")],
            consequents=[Fact(id="hypothesis_1", content="sensor_malfunction")],
            confidence=0.8,
            source="equipment_model",
        ),
        InferenceRule(
            rule_id="rule_2",
            antecedents=[Fact(id="hypothesis_1", content="sensor_malfunction")],
            consequents=[Fact(id="recommendation_1", content="replace_probe")],
            confidence=0.9,
            source="maintenance_manual",
        ),
    ]
    
    result = await self.forward_chain(facts, rules)
    return result.inferred_facts
```
